#用于兼容  For compatibility
namespace = mss_game_start
event = {
    id = mss_game_start.10
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        set_global_flag = ms_species_enabled
        if = { #美味女仆宇宙！
            limit = { any_country = { has_country_flag = ms_will_replace_galaxy_portraits_on_spawn } }
            every_galaxy_species = {
                limit = {
                    NOT = { has_species_flag = ms_immue_portrait_swap } #兼容？
                    ms_is_any_maid_portrait = no
                }
                ms_swap_to_maid_portrait = yes
                set_timed_species_flag = { flag = ms_will_change_leader_portrait days = 1 }
            }
            every_country = {
                limit = { NOT = { has_country_flag = ms_immue_portrait_swap } }
                every_owned_leader = {
                    limit = { species = { has_species_flag = ms_will_change_leader_portrait } }
                    change_leader_portrait = species
                }
                every_pool_leader = {
                    limit = { species = { has_species_flag = ms_will_change_leader_portrait } }
                    change_leader_portrait = species
                }
            }
        }
        every_playable_country = {
            limit = { is_ai = no }
            create_message = {
                type = MESSAGE_MSS_STARTUP
                days = 1
                localization = MESSAGE_MSS_STARTUP_desc
            }
            country_event = { id = mss_nomales.1 }
        }
    }
}
#无奈之举
namespace = mss_nomales
country_event = {
    id = mss_nomales.1
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        every_owned_species = {
            limit = {
                ms_is_any_maid_portrait = yes
                NOR = { species_gender = female species_gender = indeterminable }
            }
            change_species_characteristics = { gender = female }
        }
        every_owned_leader = {
            limit = {
                species = { ms_is_any_maid_portrait = yes  }
                NOR = { gender = female gender = indeterminable }
            }
            set_gender = female
        }
        every_pool_leader = {
            limit = {
                species = { ms_is_any_maid_portrait = yes  }
                NOR = { gender = female gender = indeterminable }
            }
            set_gender = female
        }
    }
}
