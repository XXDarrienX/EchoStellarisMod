#Reanimated Armies: Reanimating the enemy's army as your own
#An army has been killed in ground combat
# This = owner
# From = army
# FromFrom = opponent
# FromFromFrom = planet
country_event = {
	id = vb_psi_souleater.13
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = fromfrom
		fromfrom = {
			has_origin = origin_vb_psi_soul_eater
		}
		from = {
			exists = species
			species = {
				is_robotic = no
			}
		}
		NOT = { is_country_type = "gate_builders" }
	}

	immediate = {
		random_list = {
			50 = {
                modifier = {
                    factor = 0
                    fromfrom = { has_edict = soul_eater_ground_war }
                }
            }
			50 = {
				fromfromfrom = {
					create_army = {
						owner = root.fromfrom
						type = husk_army
						species = root.from.species
					}
				}
			}
            
		}
	}
}
