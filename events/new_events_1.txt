# Rise of the Galactic Empire
country_event = {
	id = socialist.301
	title = "socialist.301.name"
	desc = "socialist.301.normal.desc"
	picture = GFX_evt_coronation
	show_sound = event_announcement

	is_triggered_only = yes

	immediate = {
		store_galactic_community_leader_backup_data = {
			FLAG = yes
			NAME = yes
			ETHICS = yes
			GOVERNMENT = yes
		}
		if = {
			limit = {
				is_gestalt = no
				is_megacorp = no
			}
			shift_ethic = "ethic_fanatic_egalitarian"
			change_government = {
				authority = auth_democratic
				cooldown = no
				remove_invalid_civics = yes
			}
			force_add_civic = civic_galactic_chancellor
		}
		else_if = {
			limit = { is_gestalt = yes }
			force_add_civic = civic_galactic_chancellor
		}
		else_if = {
			limit = { is_megacorp = yes }
			shift_ethic = "ethic_fanatic_egalitarian"
			change_government = {
				authority = auth_democratic
				cooldown = no
				remove_invalid_civics = yes
			}
		}
		set_government_cooldown = no
		set_name = Galactic_Republic
		room_name_override = ""
		change_country_flag = {
			icon = {
				category = "zzzteesemblems7"
				file = "8flag5.dds"
			}
			background= {
				category = "backgrounds"
				file = "v.dds"
			}
			colors={
				"blue"
				"blue"
				"null"
				"null"
			}
		}
		every_playable_country = {
			limit = {
				NOT = { is_same_value = root }
			}
			country_event = { id = socialist.303 }
		}
		
	}

	option = {
		name = emperor.2.a
		add_resource = {
			influence = 250
		}
		custom_tooltip = socialist.301.a.tooltip
	}
}
