country_event = {
    id = mss_gray.20
    hide_window = yes
    is_triggered_only = yes
    trigger = { has_country_flag = mss_enable_gray_maid }
    immediate = {
        event_target:gray_country = {
            every_owned_leader = {
                ms_update_gray_portrait = yes
            }
        }
        every_owned_leader = {
            limit = { has_leader_flag = gray_leader }
            ms_update_gray_portrait = yes
        }
    }
}
