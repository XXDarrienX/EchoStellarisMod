	is_triggered_only = yes
	hide_window = yes

			from.species = { has_trait = complex_species_trait_Cyclopean }
	}

	immediate = {
		from = {
			add_trait = {
				trait = complex_admiral_trait_Blurred_Vision

			}
		}
	}
}
