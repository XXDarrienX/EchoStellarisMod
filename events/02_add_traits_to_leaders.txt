namespace = complex_traits

country_event = {
	id = complex_traits.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
			from.species = { has_trait = complex_species_trait_Cyclopean }
	}

	immediate = {
		from = {
			if = {
				limit = { leader_class = admiral }
				add_trait = complex_admiral_trait_Blurred_Vision
			}
		}
	}
}
