namespace = event_brothel


country_event = {
	id = event_brothel.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_country_type = default
		any_relation = {
    			is_country_type = default
    			has_commercial_pact = from
    			has_valid_civic = civic_mega_brothel
		}
	}

	immediate = {
		add_modifier = {
            modifier = brothel_growth_commercial_pact
            days = 1800
            mult = value:brothel_bonus
        }
    }
}
