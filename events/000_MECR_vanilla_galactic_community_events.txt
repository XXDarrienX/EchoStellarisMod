# Proposal to form the Galactic Community (HIDDEN)
event = {
	id = galcom.1
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		NOR = {
			has_global_flag = galactic_community_founded
			has_global_flag = galcom_founding_begun
		}
		count_playable_country = { # minimum 3 countries
			limit = {
				is_homicidal = no
			}
			count >= 3
		}
		any_playable_country = {
			perc_communications_with_playable >= 0.66
			is_homicidal = no
			is_gestalt = no
		}
	}

	immediate = {
		set_global_flag = galcom_founding_begun
		random_playable_country = {
			limit = {
				perc_communications_with_playable >= 0.66
				is_homicidal = no
				is_gestalt = no
			}
			country_event = { id = galcom.2 }
			every_playable_country = {
				limit = {
					NOT = { is_same_value = prev }
					has_communications = prev
					is_homicidal = no
				}
				country_event = { id = galcom.2 }
			}
		}
		random_country = {
			limit = { is_country_type = global_event }
			country_event = { id = galcom.3 days = 360 }
		}
	}
}

country_event = { #300.传统：等价交换&赫赫君威			Quid Pro Quo & Majestic Prestige
	id = galcom.300
	
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_active_tradition = tr_politics_majestic_prestige
				OR = {
					is_galactic_custodian = yes
					is_galactic_emperor = yes
				}
			}
			every_galcom_member = {
				limit = {
					position_on_last_resolution = support
					has_active_tradition = tr_politics_quid_pro_quo
					NOT = { is_same_value = root }
				}
				root = {
					add_opinion_modifier = {
						who = prev
						modifier = opinion_majestic_prestige_senate_supporter
					}
				}
			}
			every_galcom_member = {
				limit = {
					position_on_last_resolution = oppose
					NOT = { is_same_value = root }
				}
				if = {
					limit = {
						num_favors = {
							target = root
							value > 0
						}
					}
					remove_favors = {
						target = root
						value = 1
					}
					create_message = {
						type = MESSAGE_FAVOR_LOST_TRADITION
						localization = MESSAGE_FAVOR_LOST_POLITICS_TRADITION_DESC
						days = 30
						target = root.capital_scope
						variable = {
							type = name
							localization = WHO
							scope = root
						}
					}
				}
				else = {
					add_modifier = {
						modifier = majestic_prestige_senate_againster
						years = 8
					}
				}
			}
		}
		else = {
			every_galcom_member = {
				limit = {
					position_on_last_resolution = support
					has_active_tradition = tr_politics_quid_pro_quo
					NOT = {
						is_same_value = root
					}
				}
				add_favors = {
					target = prev
					value = 1
				}
				create_message = {
					type = MESSAGE_FAVOR_GAINED_TRADITION
					localization = MESSAGE_FAVOR_GAINED_POLITICS_TRADITION_DESC
					days = 30
					target = root.capital_scope
					variable = {
						type = name
						localization = WHO
						scope = root
					}
				}
			}
		}
	}
}
