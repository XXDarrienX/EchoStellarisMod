namespace = mealeaders

country_event = {
	id = mealeaders.01
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		from = { 
			species = { has_trait = trait_biggest_fish }
			Leader = { 
				NOT = { has_leader_flag = has_biggest_fish_trait } 
			}
		}
	}

	immediate = {
		from = {
			if = { 
				limit = { leader_class = ruler }
				set_leader_flag = has_biggest_fish_trait
				random_list = {
					1 = { add_trait = trait_ruler_sea_dragon }
					6 = { }
				} 
				break = yes
			}
			if = { 
				limit = { leader_class = governor }
				set_leader_flag = has_biggest_fish_trait
				random_list = {
					1 = { add_trait = trait_governor_sea_dragon }
					7 = { }
				} 
				break = yes
			}
			if = { 
				limit = { leader_class = general }
				set_leader_flag = has_biggest_fish_trait
				random_list = {
					1 = { add_trait = trait_general_leviathan }
					5 = { }
				} 
				break = yes
			}
		}
	}
}
