# Check all planet from all empires for branch offices which owner have Gospel of the masses civic AND the temple built

}

# Try to turn every preachers, subversive or not, of said planets into spiritualists
planet_event = {
	id = underworld_realms.2
	hide_window = yes

	is_triggered_only = yes
	
	immediate = {
		every_owned_pop = {
			limit = {
				AND = {
					NOT = { pop_has_ethic = ethic_spiritualist }
					OR ={
						has_job = preacher
						has_job = subversive_preacher
					}
				}
			}
			random_list = {
				11 = {}
				1 = {
					pop_change_ethic = ethic_spiritualist
				}
			}
		}
	}
}

# Crime Raid edict
country_event = {
	id = underworld_realms.100
	title = underworld_realms.100.name
	desc = underworld_realms.100.desc
	picture = GFX_evt_dark_alley
	show_sound = event_criminal_activity
	
	is_triggered_only = yes
	
	immediate = {
		save_event_target_as = filthy_criminal
		country_event = { id = underworld_realms.103 days = 3600 }
		every_country = {
			limit = {
				is_country_type = default
			}
			
			every_owned_planet = {
				limit = {
					has_branch_office = yes
				}
				
				add_modifier = { modifier = "crime_raid" days = 720 }
				
				owner = {
					add_opinion_modifier = {
						who = ROOT
						modifier = opinion_crime_raid
					}
				}
				
				owner = { set_country_flag = is_targeted_by_crime_raid }
			}
		}
		
		every_country = {
			limit = {
				has_country_flag = is_targeted_by_crime_raid
			}
			
			remove_country_flag = is_targeted_by_crime_raid
			country_event = { id = underworld_realms.101 days = 10 random = 3 }
		}
	}
	
	option = {
		trigger = {
			has_civic = civic_criminal_heritage
		}
		
		name = underworld_realms.100.a
		custom_tooltip = underworld_realms.100.a.tooltip
	}
}

# Ceremony Broadcast
country_event = {
	id = underworld_realms.200
	title = underworld_realms.200.name
	desc = underworld_realms.200.desc
	picture = GFX_evt_temple_of_prosperity
	show_sound = event_celebration
	location = root
	
	is_triggered_only = yes

	trigger = {
		AND = {
			any_country = {
				any_owned_planet = {
					has_branch_office = yes
					has_holding = {
						owner = planet.branch_office_owner
						Holding = building_temple_of_prosperity
					}
				}
			}
		}
	}
	
	immediate = {
		save_event_target_as = gospel_country
		country_event = { id = underworld_realms.204 days = 3600 }
		
		every_country = {
			limit = {
				is_country_type = default
			}
			
			every_owned_planet = {
				limit = {
					has_branch_office = yes
					has_holding = {
						owner = planet.branch_office_owner
						Holding = building_temple_of_prosperity
					}
				}
				
				owner = { set_country_flag = is_gospeled }
				
				add_modifier = { modifier = "ceremony_broadcast_branch_office" days = 720 }
			}
		}
		
		every_country = {
			limit = {
				has_country_flag = is_gospeled
			}
			
			if = {
				limit = {
					this = {
						OR = {
							has_ethic = ethic_materialist
							has_ethic = ethic_fanatic_materialist
						}
					}
				}
				
				country_event = { id = underworld_realms.203 days = 10 random = 3 }
			}
			else_if = {
				limit = {
					this = {
						OR = {
							has_ethic = ethic_spiritualist
							has_ethic = ethic_fanatic_spiritualist
						}
					}
				}
				
				country_event = { id = underworld_realms.202 days = 10 random = 3 }
			}
			else = {
				country_event = { id = underworld_realms.201 days = 10 random = 3 }
			}
			
			remove_country_flag = is_gospeled
		}
	}
	
	option = {
		trigger = {
			has_valid_civic = civic_gospel_of_the_masses
		}
		name = underworld_realms.200.a
		
		add_modifier = { modifier = ceremony_broadcast_country days = 3600 }
	}
}

	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		random_playable_country = {
			limit = {
				is_country_type = default
				any_owned_planet = {
					has_branch_office = yes
					branch_office_owner = {
						is_criminal_syndicate = yes
							OR = {
								has_policy_flag = policy_crime_mafia_cartel
								has_policy_flag = policy_crime_political_activists
								has_policy_flag = policy_crime_hacking_cell
								has_policy_flag = policy_crime_terrorist
								has_policy_flag = policy_crime_assassins_order
								has_policy_flag = policy_crime_subversive_cult
						}
					}
					OR = {
						AND = {
							exists = event_target:crime_target.overlord
							NOT = { branch_office_owner = { is_same_value = event_target:crime_target.overlord } }
						}
						NOT = { exists = event_target:crime_target.overlord }
					}
					planet_crime >= 25
				}
			}
			
			this = { save_event_target_as = crime_target }
			
			random_owned_planet = {
				limit = {
					has_branch_office = yes
					branch_office_owner = {
						is_criminal_syndicate = yes
					}
					OR = {
						AND = {
							exists = event_target:crime_target.overlord
							NOT = { branch_office_owner = { is_same_value = event_target:crime_target.overlord } }
						}
						NOT = { exists = event_target:crime_target.overlord }
					}
					OR = {
						NOT = { branch_office_owner = { is_in_federation_with = owner } }
						AND = {
							branch_office_owner = {
								federation = { has_federation_type = crime_federation }
								is_federation_leader = yes
							}
							owner = { federation = { has_federation_type = crime_federation } }
						}
					}
					planet_crime >= 25
					NOT = { has_modifier = crime_lord_deal }
					NOT = { has_planet_flag = crime_recently_targeted }
				}
				
				branch_office_owner = { save_event_target_as = office_owner }
				this = { save_event_target_as = target_planet }
				
				# Crime is between 25 and 50
				if = {
					limit = {
						AND = {
							planet_crime >= 25
							planet_crime < 50
						}
					}
					
					random_list = {
						60 = {}
						40 = {
							event_target:target_planet = { set_timed_planet_flag = { flag = crime_recently_targeted days = 360 } }
							event_target:office_owner = { country_event = { id = underworld_realms.1010 days = 1 random = 30 } }
						}
					}
				}
				
				# Crime is between 50 and 99
				else_if = {
					limit = {
						AND = {
							planet_crime >= 50
							planet_crime < 99
						}
					}
					
					if = {
						limit = {
							event_target:target_planet = { is_capital = no }
							num_pops >= 50
						}
						
						random_list = {
							60 = {}
							40 = {
								event_target:target_planet = { set_timed_planet_flag = { flag = crime_recently_targeted days = 720 } }
								random_list = {
									66 = { event_target:office_owner = { country_event = { id = underworld_realms.1010 days = 1 random = 30 } } }
									33 = { event_target:office_owner = { country_event = { id = underworld_realms.1050 days = 1 random = 30 } } }
									12 = { event_target:office_owner = { country_event = { id = underworld_realms.1100 days = 1 random = 30 } } }
								}
							}
						}
					}
					
					else = {
						random_list = {
							60 = {}
							40 = {
								event_target:target_planet = { set_timed_planet_flag = { flag = crime_recently_targeted days = 720 } }
								random_list = {
									66 = { event_target:office_owner = { country_event = { id = underworld_realms.1010 days = 1 random = 30 } } }
									33 = { event_target:office_owner = { country_event = { id = underworld_realms.1050 days = 1 random = 30 } } }
								}
							}
						}
					}
				}
				
				# Crime is 100
				else_if = {
					limit = {
						planet_crime = 100
					}
					
					if = {
						limit = {
							event_target:target_planet = { is_capital = no }
							num_pops >= 50
						}
						
						event_target:target_planet = { set_timed_planet_flag = { flag = crime_recently_targeted days = 720 } }
						random_list = {
							20 = { event_target:office_owner = { country_event = { id = underworld_realms.1100 days = 1 random = 30 } } }
							50 = { event_target:office_owner = { country_event = { id = underworld_realms.1010 days = 1 random = 30 } } }
							50 = { event_target:office_owner = { country_event = { id = underworld_realms.1050 days = 1 random = 30 } } }
						}
					}
					
					else = {
						event_target:target_planet = { set_timed_planet_flag = { flag = crime_recently_targeted days = 720 } }
						event_target:office_owner = { country_event = { id = underworld_realms.1050 days = 1 random = 30 } }

# Shroud the planet !
planet_event = {
	id = underworld_realms.3050
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = {
				is_country_type = shroud
			}
			
			country_event = { id = shroud_criticality.1 }
		}
			
		save_event_target_as = poor_doomed_planet
	
		# warn author
		branch_office_owner = {
			save_event_target_as = office_owner
			
			country_event = { id = underworld_realms.3024 }
		}
		
		# warn others
		every_country = {
			limit = {
				NOR = {
					is_same_value = event_target:poor_doomed_planet.branch_office_owner
					is_same_value = event_target:poor_doomed_planet.owner
				}
				AND = {
					has_communications = event_target:poor_doomed_planet.branch_office_owner
					has_communications = event_target:poor_doomed_planet.owner
				}
			}
			country_event = { id = underworld_realms.3026 }
		}
		
		owner = {
			# warn former owner
			country_event = { id = underworld_realms.3025 }
		}
		
		if = {
			limit = {
				OR = {
					is_active_resolution = "resolution_rulesofwar_independent_tribunals"
					is_active_resolution = "resolution_rulesofwar_last_resort_doctrine"
					is_active_resolution = "resolution_rulesofwar_demobilization_initiative"
				}
			}
			event_target:poor_doomed_planet.branch_office_owner = {
				set_timed_country_flag = { flag = resolution_breached_eldritched_planet days = 3600 }
			}
		}
		
		if = {
			limit = {
				exists = owner
				owner = {
					NOT = { is_same_value = event_target:poor_doomed_planet.branch_office_owner }
				}
			}
			# Generate threat
			add_threat = { who = event_target:poor_doomed_planet.branch_office_owner amount = 3 }
			# modifier for allies + those upset by conversion
			every_country = {
				limit = {
					NOR = {
						is_same_value = event_target:poor_doomed_planet.branch_office_owner
						is_same_value = root.owner
						AND = {
							has_federation = yes
							is_in_federation_with = event_target:poor_doomed_planet.branch_office_owner
						}
					}
					AND = {
						has_communications = event_target:poor_doomed_planet.branch_office_owner
						has_communications = root.owner
					}
					OR = {
						AND = {
							has_federation = yes
							is_in_federation_with = root.owner
						}
						has_ai_personality = awakened_fallen_empire_materialist
						AND = {
							is_country_type = default
							OR = {
								is_materialist = yes
								is_egalitarian = yes
								has_authority = auth_machine_intelligence
							}
						}
					}
				}
				add_opinion_modifier = {
					modifier = opinion_shrouded_a_world
					who = event_target:poor_doomed_planet.branch_office_owner
				}
			}
			# modifiers for victim
			owner = {
				add_opinion_modifier = {
					modifier = opinion_shrouded_my_world
					who = event_target:poor_doomed_planet.branch_office_owner
				}
				if = {
					limit = { NOT = { has_ethic = ethic_gestalt_consciousness } }
					add_modifier = {
						modifier = colossus_victim
						days = 10800 # 30 years
					}
				}
			}
			# Add war exhaustion to planet owner 
			owner = {
				add_static_war_exhaustion = {
					attacker = event_target:poor_doomed_planet.branch_office_owner
					location = root
					value_for_planet_destruction = 1.0
				}
			}
		}
		every_owned_pop = {
			kill_pop = yes
		}
		
		remove_all_buildings = yes
		
		# Special graphics for Habitats and Ring Worlds
		if = {
			limit = {
				gpm_is_planet_class_habitat = yes
			}
			spawn_habitat_cracker_effect = yes
			remove_planet = yes
		}
		else_if = {
			limit = {
				is_planet_class = pc_ringworld_habitable
			}
			spawn_ringworld_cracker_effect = yes
			change_pc = pc_ringworld_habitable_damaged
		}
		
		change_pc = pc_shrouded
		
		clear_deposits = yes
	}
}
