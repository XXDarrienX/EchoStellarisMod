namespace = meaorigins

event = {
	id = meaorigins.01
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		any_country = {
			has_origin = origin_deep_dwellers
		}
		NOT	= {
			has_global_flag = randomized_depths_flag
		} 
	}

	immediate = {
		every_system = {
			limit = {
				NOT = {
					exists = owner
				}
			}
			every_system_planet = {
				limit = {
					NOT = {
						exists = owner
					}
					is_colonizable = yes
					NOR = {
						is_planet_class = pc_nuked
						is_planet_class = pc_habitat
						is_planet_class = pc_relic
						is_planet_class = pc_city
						is_ringworld = yes
					}
				}
				random_list = {
					15 = {}
					1 = {

						modifier = {
							factor = 5
							OR = {
								is_planet_class = pc_gaia
								is_planet_class = pc_ocean
								is_planet_class = pc_arctic
							}
						}
					
						modifier = {
							factor = 3
							has_climate = wet
						}
						modifier = {
							factor = 0.25
							has_climate = dry
						}

						modifier = {
							factor = 2
							planet_size > 24
						}
						modifier = {
							factor = 2
							planet_size > 22
						}
						modifier = {
							factor = 2
							planet_size > 20
						} 
						modifier = {
							factor = 0.75
							planet_size < 17
						}
						modifier = {
							factor = 0.5
							planet_size < 15
						}
						modifier = {
							factor = 0.25
							planet_size < 13
						}

						if = {
							limit = {
								is_planet_class = pc_alpine
								NOR = {
									has_modifier = deep_ravines_mod
									has_modifier = crushing_depths_mod
								} 
							}
							add_modifier = {
								modifier = deep_ravines_mod
							}
						}
						if = {
							limit = {
								NOR = {
									has_modifier = deep_ravines_mod
									has_modifier = crushing_depths_mod
								} 
							}
							add_modifier = {
								modifier = crushing_depths_mod
							}
						}
				
					}
				}
			}
			every_system_planet = {
				limit = {
					NOT = {
						exists = owner
					}
					NOT = {
						has_modifier = crushing_depths_mod
					}
					is_colonizable = yes
					NOR = {
						is_planet_class = pc_gaia
						is_planet_class = pc_ocean
						is_planet_class = pc_nuked
						is_planet_class = pc_habitat
						is_planet_class = pc_relic
						is_planet_class = pc_city
						is_ringworld = yes
					}
				}
				random_list = {
					10 = {
					}
					1 = {

						modifier = {
							factor = 10
							is_planet_class = pc_alpine
						}

						modifier = {
							factor = 3
							has_climate = dry
						}
						modifier = {
							factor = 0.25
							has_climate = wet
						}

						modifier = {
							factor = 2
							planet_size > 23
						}
						modifier = {
							factor = 2
							planet_size > 21
						}
						modifier = {
							factor = 2
							planet_size > 19
						} 
						modifier = {
							factor = 0.75
							planet_size < 17
						}
						modifier = {
							factor = 0.5
							planet_size < 15
						}
						modifier = {
							factor = 0.25
							planet_size < 13
						}

						if = {
							limit = {
								is_planet_class = pc_arctic
								NOR = {
									has_modifier = deep_ravines_mod
									has_modifier = crushing_depths_mod
								} 
							}
							add_modifier = {
								modifier = crushing_depths_mod
							}
						}
						if = {
							limit = {
								NOR = {
									has_modifier = deep_ravines_mod
									has_modifier = crushing_depths_mod
								} 
							}
							add_modifier = {
								modifier = deep_ravines_mod
							}
						}
				
					}
				}
			}
		}
		set_global_flag = randomized_depths_flag
	}
}

country_event = {
	id = meaorigins.02
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		NOT	= {
			has_country_flag = set_depths_flag
		} 
		has_origin = origin_deep_dwellers
	}

	immediate = {
		every_owned_planet = {
			if = {
				limit = {
					NOT = {
						has_modifier = crushing_depths_mod
					} 
					any_owned_pop_group = {
						has_trait = trait_deep_dwellers
					}
				}
				set_planet_size = 30
				add_modifier = {
					modifier = crushing_depths_mod
				}
			}
		}
		set_country_flag = set_depths_flag

	}

}

planet_event = {
	id = meaorigins.03
	hide_window = yes
	is_triggered_only = yes
		
	trigger = {
		OR	= {
			has_modifier = crushing_depths_mod
			has_modifier = deep_ravines_mod
		}
	}

	immediate = {
		
		if = {
			limit = {
				has_modifier = crushing_depths_mod
				NOT	= {
					is_planet_class = pc_ocean
				}
			}
			remove_modifier = crushing_depths_mod
			add_modifier = {
				modifier = deep_ravines_mod
			}
		}

		if = {
			limit = { 
				has_modifier = deep_ravines_mod
				is_planet_class = pc_ocean
			}
			remove_modifier = deep_ravines_mod
			add_modifier = {
				modifier = crushing_depths_mod
			}
		}

	}

}
