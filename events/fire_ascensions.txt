# Fuel For The Pyre 
country_event = {
	id = fiery.01
	title = "fiery.01.name"
	desc = "fiery.01.desc"
	picture = GFX_fire_perk_event
	show_sound = event_solar_fusion
	
	is_triggered_only = yes
	
	immediate = {
		every_owned_pop_group = {
			limit = {
				is_same_species = root
				NOT = { species = { has_trait = trait_fire_perk1 } }
			}
			modify_species = {
				species = this
				add_trait = trait_fire_perk1
			}
			species = { save_event_target_as = fiery_species }
		}
		change_dominant_species = { species = event_target:fiery_species change_all = yes }
		observer_event = { id = observer.5 }
		#every_country = {
		#	limit = { 
		#		is_ai = no
		#		has_communications = root
		#		NOT = { is_same_value = root }
		#	}
		#	country_event = { id = fiery.02 }
		#}
	}
	
	option = {
		name = "fiery.01.a"
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						is_country_type = default
						is_fallen_empire = yes
					}
				}
				every_owned_leader = {
					limit = {
						is_same_species = root
					}
					add_trait = leader_trait_fire_perk1
				}
				every_pool_leader = {
					limit = {
						is_same_species = root
					}
					add_trait = leader_trait_fire_perk1
				}
			}
		}
	}
}

# Ignite The Inferno
country_event = {
	id = fiery.03	#utopia.2650
	title = "fiery.03.name"
	desc = "fiery.03.desc"
	picture = GFX_fire_perk_event
	show_sound = event_solar_fusion
	
	is_triggered_only = yes

	immediate = {
		every_owned_pop_group = {
			limit = {
				is_same_species = root
				has_trait = trait_fire_perk1
			}
			modify_species = {
				species = this
				remove_trait = trait_fire_perk1
				add_trait = trait_fire_perk2
			}
			species = { save_event_target_as = fiery_species }
		}
		change_dominant_species = { species = event_target:fiery_species change_all = yes }
	}
	option = {
		name = "fiery.03.a"
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						is_country_type = default
						is_fallen_empire = yes
					}
				}
				every_owned_leader = {
					limit = {
						is_same_species = root
					}
					add_trait = leader_trait_fire_perk2
				}
				every_pool_leader = {
					limit = {
						is_same_species = root
					}
					add_trait = leader_trait_fire_perk2
				}
			}
		}
	}
}
