country_event = {
	id = addtrait_leader.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
			from.species = { has_trait = trait_additrait_hateful }
	}
	
	immediate = {
		from = {
		switch = {
			trigger = leader_class
			ruler = {
				add_ruler_trait = trait_additrait_hateful_rul
			}
			scientist = {
				add_ruler_trait = trait_additrait_hateful_rul
			}
			governor = {
				add_ruler_trait = trait_additrait_hateful_rul
			}
			general = {
				add_ruler_trait = trait_additrait_hateful_rul
			}
			admiral = {
				add_ruler_trait = trait_additrait_hateful_rul
				}
			}
		}
	}
}
