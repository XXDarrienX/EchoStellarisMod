namespace = addtrait_leader

country_event = {
	id = addtrait_leader.2
	is_triggered_only = yes
	hide_window = yes

	trigger = {
			from.species = { has_trait = trait_additrait_born_warriors }
	}

	immediate = {
		from = {
		switch = {
			trigger = leader_class
			admiral = {
				add_trait = trait_additrait_born_warriors_adm
			}
			general = {
				add_trait = trait_additrait_born_warriors_gen
				}
			}
		}
	}
}
