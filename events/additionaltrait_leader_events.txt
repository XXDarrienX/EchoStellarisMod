country_event = {
	id = addtrait_leader.2
	is_triggered_only = yes
	hide_window = yes

			from.species = { has_trait = trait_additrait_born_warriors }
	}

	immediate = {
		from = {
			add_trait = {
				trait = leader_additrait_born_warriors
				show_message = no
			}
		}
	}
}
