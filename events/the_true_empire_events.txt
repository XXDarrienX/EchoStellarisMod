# Event starter
country_event = {
	id = the_true_empire.0
	hide_window = yes
	
	# Trigger on global_event for the duration between 10 years in midgame and 0 years past endgame
	trigger = {
		mid_game_years_passed >= 0
		is_country_type = global_event
		NOT = { has_global_flag = fallen_admiral_crisis_started }
		any_country = {
			is_country_type = fallen_empire
			is_fallen_empire_machine = no
		}
	}
	
	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 500
			mid_game_years_passed >= 5
		}
		modifier = {
			factor = 5
			count_country = {
				limit = { is_country_type = fallen_empire }
				count >= 3
			}
		}
	}
	
	immediate = {
		set_global_flag = fallen_admiral_crisis_started
		random_country = {
			limit = { 
				is_country_type = fallen_empire
				is_fallen_empire_machine = no
			}
			save_global_event_target_as = the_original_empire
			set_country_flag = the_original_empire
			random_owned_fleet = {
				limit = { is_mobile = yes   fleet_size >= 50 }
				delete_fleet = this
			}
			country_event = {
				id = the_true_empire.10
				days = 360
				random = 30
			}
		}
		every_playable_country = {
			country_event = {
				id = the_true_empire.1
				days = 30
			}
		}
	}
}
