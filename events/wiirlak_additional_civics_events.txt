	}
}

	fire_only_once = yes

		has_civic = wiirlak_ac_civic_investment
	}

		name = wiirlak_ac_event_b
		add_resource = {
			energy = 500
			minerals = 250
			alloys = 100
			influence = 100
		}

			energy = 1500
			minerals = 750
			alloys = 250
			influence = 250
		}

	}
}

# Nyctophobia Origin
# On start - Add the Penal Colonies tech & modifies the original moon into a jungle penal colony
country_event = {
	id = wiirlak_ac.3020
	hide_window = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_origin = wiirlak_ac_origin_nyctophobia
	}
	
	immediate = {
		give_technology = { message = yes tech = tech_penal_colonies }
		capital_scope = {
			random_moon = {
				change_pc = pc_tropical
				add_deposit = d_tropical_island
				add_deposit = d_lush_jungle
				add_deposit = d_fertile_lands
				
				add_deposit = d_geothermal_vent
				add_deposit = d_rushing_waterfalls

				add_deposit = d_mineral_fields
				add_deposit = d_submerged_ore_veins

				add_deposit = d_dense_jungle
				add_deposit = d_dense_jungle
				add_deposit = d_dense_jungle
				add_deposit = d_toxic_kelp
				add_deposit = d_noxious_swamp

				create_colony = { owner = ROOT }
				create_pop = { species = ROOT }
				add_modifier = { modifier = penal_colony }
				remove_deposit = d_energy_5
			}
		}
	}
}

# Starless
# At the Start - Replace the start by a Black Hole, change the planet size and replace the deposits (+1 Crystal / Mote)
country_event = {
	id = wiirlak_ac.3110
	hide_window = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_origin = wiirlak_ac_origin_starless
	}
	
	immediate = {
		
		solar_system = {
			system_star = {
				change_pc = pc_black_hole
			}
			every_system_planet = {
				every_moon = {
					change_pc = pc_barren_cold
				}
			}
			every_system_planet = {
				random_moon = {
					change_pc = pc_frozen
				}
				random_moon = {
					change_pc = pc_frozen
				}
			}
		}

		capital_scope = {
			set_planet_size = 20

			remove_deposit = d_forgiving_tundra
			remove_deposit = d_forgiving_tundra
			remove_deposit = d_forgiving_tundra
			remove_deposit = d_fungal_forest
			remove_deposit = d_fungal_forest
			remove_deposit = d_fungal_forest
			remove_deposit = d_fungal_caves
			remove_deposit = d_fungal_caves
			remove_deposit = d_fungal_caves

			add_deposit = d_geothermal_vent
			add_deposit = d_frozen_gas_lake
			add_deposit = d_rich_mountain
			add_deposit = d_ore_rich_caverns
			add_deposit = d_mineral_fields
			add_deposit = d_crystalline_caverns
			add_deposit = d_dust_caverns
			add_deposit = d_fungal_caves
		}
	}
