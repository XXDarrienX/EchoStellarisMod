# via on_survey = {} | this = surveying vessel
ship_event = {
	id = UCmod_precursor.1
	title = "UCmod_precursor.1.name"
	desc = "UCmod_precursor.1.desc"
	picture = GFX_evt_outpost
	show_sound = event_default
	location = from

	fire_only_once = yes

	is_triggered_only = yes

	trigger = {
		OR = {
			AND = {
				NOT = { has_global_flag = Precursor_Selection_installed }
				owner = {
					is_ai = no
					OR = {
						has_precursor_intro = no
						AND = {
							has_origin = origin_endangered
							NOT = { has_country_flag = UCmod_volgan_started }
						}
					}
				}
			}
			AND = {
				has_global_flag = Precursor_Selection_installed
				owner = {
					is_ai = no
					OR = {
						AND = {
							NOT = { has_country_flag = volgan_intro }
							has_country_flag = picked_volgan
						}
						AND = {
							has_origin = origin_endangered
							NOT = { has_country_flag = UCmod_volgan_started }
						}
					}
				}
			}
		}
		solar_system = {
			has_star_flag = precursor_volgan_1
			NOR = {
				is_same_value = root.owner.capital_scope.solar_system
				has_star_flag = sol_system
			}
		}
		OR = {
			NOT = { has_global_flag = volgan_started }
			owner = {
				AND = {
					has_origin = origin_endangered
					NOT = { has_country_flag = UCmod_volgan_started }
				}
			}
		}
		FROM = {
			has_anomaly = no
			is_star = no
			NOT = { exists = archaeological_site }
			NOT = { has_planet_flag = suppress_archaeological_sites }
			is_planet_class = pc_frozen
		}
	}
	weight_multiplier = {
		factor = 1
		modifier = {
			factor = @origin_shoulders_multiplier
			owner = {
				has_origin = origin_shoulders_of_giants
				has_country_flag = origin_shoulders_closure
			}
		}
		modifier = {
			factor = 10
			owner = {
				has_origin = origin_endangered
				#has_country_flag = origin_endangered_closure
			}
		}
		modifier = {
			factor = 1000
			owner = {
				has_country_flag = picked_volgan
			}
		}
		modifier = {
			factor = 0
			owner = {
				has_country_flag = PreSelect_Recent_Discovery
			}
		}
	}

	immediate = {
		set_global_flag = volgan_started
		owner = { set_country_flag = volgan_intro }
		owner = { set_country_flag = vultaum_intro }
		if = {
			limit = {
				owner = {
					has_origin = origin_endangered
				}
			}
			owner = { set_country_flag = UCmod_volgan_started }
		}
		solar_system = { set_star_flag = first_volgan_site@root.owner }
	}

	option = {
		name = CURIOUS
		hidden_effect = {
			owner = { country_event = { id = story.5 days = 30 } }
			from = { create_archaeological_site = volgan_digsite_1 }
		}
	}
}

						}

country_event = {
	id = UCmod_precursor.20
	hide_window = yes

	fire_only_once = yes

	is_triggered_only = yes

	immediate = {
		set_global_flag = volgan_home_system_discovered
		set_country_flag = volgan_home_system
		if = {
			limit = {
				any_system_within_border = { has_star_flag = precursor_volgan_1 }
			}
			random_system_within_border = {
				limit = { has_star_flag = precursor_volgan_1 }
				save_event_target_as = volgan_home_spawn_system
			}
		}
		else = {
			random_system_within_border = {
				save_event_target_as = volgan_home_spawn_system
			}
		}
