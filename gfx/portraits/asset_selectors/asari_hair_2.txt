	default = "gfx/models/portraits/asari/asari_hair_2.dds"
	
	game_setup = {#will run with a limited country scope. species and government is set but the country does not actually exist
		default = "gfx/models/portraits/asari/asari_hair_2.dds"
	}

	#species scope
	species = { #generic portrait for a species
		default = "gfx/models/portraits/asari/asari_hair_2.dds"
	}
	
	#pop scope
	pop = { #for a specific pop
	
		default = "gfx/models/portraits/asari/asari_hair_2.dds"

		random = {
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			
			}
		}	
	}
	
	
	
	
	#leader scope
	leader = { #scientists, generals, admirals, governor	
		
		random = {
			trigger = { leader_class = commander }
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			}
		}	
		
		random = {
			trigger = { leader_class = scientist }
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"

			}
		}		
		
		random = {
			trigger = { leader_class = official }
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			}
		}
	}
  
	#leader scope 
	ruler = { #for rulers
		
		random = {
			trigger = { 
				owner = { has_militarist_government = yes }
			}
				
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			}
		}		
		
		random = {
			trigger = {
				OR = {
					owner = { has_spiritualist_government = yes } 
				}
			}	
			
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			}
		}		
		
		random = {
			trigger = {
                owner = {			
				OR = {
					    has_government = gov_stagnant_ascendancy
						has_primitive_government = yes
						has_materialist_government = yes
						has_pacifist_government = yes 
					
				}
				}
			}
			
			list = {
				"gfx/models/portraits/asari/asari_hair_2.dds"
			}
		}
	}
}
