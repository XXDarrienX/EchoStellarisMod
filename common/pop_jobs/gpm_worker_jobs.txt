			factor = 50

			}
		modifier = {
			weight = 20

			owner = { is_at_war = yes }
		}
		modifier = {
			factor = 3
			has_trait = trait_resilient
	}
		modifier = {
			factor = 3
			planet = { planet_crime > 0 }

			factor = 2

		}
		modifier = {

		}
		modifier = {
			factor = 1.1

				has_trait = trait_traditional
