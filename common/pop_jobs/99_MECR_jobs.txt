shattered_planet_miner = {
	category = worker

	swappable_data = {
		default = {
			icon = miner
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_shattered_planet_mining_hub
		}
	}

	tags = { minerals }

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_worker_job
	possible = {
		hidden_trigger = { exists = owner }
		owner = {
			is_gestalt = no
		}
	}

	resources = {
		category = planet_miners
		produces = {
			minerals = 6
		}
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_material_requisitorium
				}
			}
			minerals = @requisitorium_minerals_reduction
		}
		upkeep = {
			energy = 2
		}
		# NOT EFFECTED BY PLANET DEPOSITS ETC
		produces = {
			energy = -1
			mult = modifier:planet_miners_energy_produces_add
		}
		produces = {
			alloys = -1
			mult = modifier:planet_miners_alloys_produces_add
		}
		produces = {
			volatile_motes = -1
			mult = modifier:planet_miners_volatile_motes_produces_add
		}
		produces = {
			exotic_gases = -1
			mult = modifier:planet_miners_exotic_gases_produces_add
		}
		produces = {
			rare_crystals = -1
			mult = modifier:planet_miners_rare_crystals_produces_add
		}
		produces = {	# d_project_cornucopia
			trigger = {
				planet = { has_deposit = d_project_cornucopia }
				owner = {
					is_galactic_community_member = yes
					is_active_resolution = resolution_custodian_galactic_department_of_exploit
				}
			}
			minerals = -1
		}
		produces = {	# zone_betharian
			trigger = {
				planet = { num_zones = { type = zone_betharian value > 0 } }
			}
			minerals = 2
		}
		produces = {	# d_lava_tubes
			trigger = {
				planet = { has_deposit = d_lava_tubes }
			}
			minerals = -3
		}
		upkeep = {		# d_lava_tubes
			trigger = {
				planet = { has_deposit = d_lava_tubes }
			}
			energy = -0.5
		}
		produces = {	# molten_mineral_rivers
			trigger = {
				planet = { has_modifier = molten_mineral_rivers }
			}
			minerals = -1
		}
		produces = {	# ship_junkyard
			trigger = {
				planet = { has_modifier = ship_junkyard }
			}
			minerals = 2
			engineering_research = -1
		}
		produces = {	# metal_planet
			trigger = {
				planet = { has_modifier = metal_planet }
			}
			minerals = -3
		}
		produces = {	# subterranean_wildlife
			trigger = {
				planet = { has_modifier = subterranean_wildlife }
			}
			society_research = -1
		}
	}

	overlord_resources = {
		category = planet_requisitioned_minerals
		produces = {
			trigger = {
				planet = {
					has_planet_flag = has_material_requisitorium
				}
			}
			minerals = @requisitorium_minerals_overlord
		}
	}

	weight = {
		weight = @worker_job_weight
		mult = value:job_weights_modifier|JOB|miner|RESOURCE|minerals|
		modifier = {
			factor = 10
			is_enslaved = yes
			can_take_servant_job = no
			NOT = { has_slavery_type = { type = slavery_indentured } }
		}
		modifier = {
			factor = 3
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { has_technology = tech_droid_workers }
		}
		modifier = {
			factor = 200
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { NOT = { has_technology = tech_droid_workers } }
		}
		modifier = {
			factor = 0.25
			can_take_servant_job = yes
		}

		# low income modifier
		modifier = {
			factor = 3
			exists = planet
			planet = {
				has_available_jobs = miner
			}
			exists = owner
			owner = {
				has_monthly_income = {
					resource = minerals
					value < 10
				}
			}
		}
		modifier = {
			factor = 5
			has_trait = trait_cyborg_creed_of_labor
		}
	}

	inline_script = "jobs/automodding_priority_minerals"

	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = worker
				AMOUNT = local_pop_amount
			}
		}
	}
}
