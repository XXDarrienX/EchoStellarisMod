guild_courtesan = {
	category = ruler
	swappable_data = {
		default = {
			condition_string = RULER_JOB_TRIGGER
			building_icon = building_underground_clubs
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_ruler_job

	resources = {
		category = planet_courtesans
		produces = {
			unity = 6
			trade = 7
		}
		upkeep = {
			consumer_goods = 2
		}
	}

	planet_modifier = {
		pop_happiness = 0.05
		planet_stability_add = 8
	}

	### Without Edict, .5 Influence
	triggered_country_modifier = {
		potential = {

			## Eliminate double stack of influence bonus, this is WITH OUT the relevant Courtesan edict
			ROOT.owner = {
				NOT = {
					has_edict = voluntas_meretrix
				}
			}
			ROOT.planet = {
				planet_stability >= 70
			}
		}
		country_base_influence_produces_add = 0.5
	}

	### With Edict, 1 Influence
	triggered_country_modifier = {
		potential = {

			## Eliminate double stack of influence bonus, this is WITH the relevant Courtesan edict
			ROOT.owner = {
				has_edict = voluntas_meretrix
			}
			ROOT.planet = {
				planet_stability >= 70
			}
		}
		country_base_influence_produces_add = 1.0
	}
	
	weight = {
		weight = @ruler_job_weight
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 2
			has_living_standard = { type = living_standard_utopian }
		}
		modifier = {
			factor = 3
			has_trait = trait_charismatic
			has_trait = trait_robot_domestic_protocols
		}
		modifier = {
			factor = 2
			OR = {
				has_trait = trait_robot_propaganda_machines
				has_trait = trait_traditional
				has_trait = trait_psionic
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				has_trait = trait_latent_psionic
				has_trait = trait_social_pheromones
			}
		}
		modifier = {
			factor = 0.65
			has_trait = trait_repugnant
		}
		modifier = {
			factor = 1.25
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			factor = 0.75
			has_trait = trait_repugnant
		}
		modifier = {
			factor = 1.5
			has_trait = trait_necrophage
		}
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
		}
	}
}
