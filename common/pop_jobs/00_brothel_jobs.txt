pleasure_worker = {
	category = worker

	swappable_data = {
		default = {
			condition_string = WORKER_JOB_TRIGGER
			building_icon = building_hyper_entertainment_forum
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_worker_job
	possible = {
		owner = { is_fallen_empire = no }
		NOT = { has_trait = trait_mechanical }
	}

	planet_modifier = {
		logistic_growth_mult = 0.0125
		planet_jobs_produces_mult = -0.002
	}

	resources = {
		category = planet_jobs
		produces = {
			unity = 1
			trade = 3
		}
		upkeep = {
			consumer_goods = 0.5
		}
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 500
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	weight = {
		weight = @worker_job_weight
		mult = value:job_weights_modifier|JOB|entertainer|RESOURCE|unity|FACTOR|0.25|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
		}
		modifier = {
			factor = 3
			has_trait = trait_robot_domestic_protocols
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
		}
		modifier = {
			factor = 0.1
			is_domestic_servant = yes
		}

		modifier = {
			factor = 50
		}

		modifier = {
			add = 500
			exists = planet
			planet = {
				has_modifier = artistic_boom
			}
		}
	}

	inline_script = "jobs/automodding_priority_amenities"

	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = worker
				AMOUNT = local_pop_amount
			}
		}
	}
}
