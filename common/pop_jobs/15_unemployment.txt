ruler_unemployment = {
	category = ruler_unemployment
	count_as_available_for_ai = yes

	swappable_data = {
		default = {
			condition_string = RULER_JOB_TRIGGER
			icon = unemployment
			building_icon = building_capital
		}
	}

	possible = {
		hidden_trigger = { exists = owner }
		owner = {
			is_gestalt = no
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	resources = {
		category = unemployment_resources
		# Unemployment Benefits Crime event
		upkeep = {
			trigger = {
				planet = {
					has_modifier = unemployment_benefits
				}
			}
			consumer_goods = 2
		}
		upkeep = {
			trigger = {
				planet = {
					has_modifier = activity_program
				}
			}
			unity = 1
		}
		upkeep = {
			trigger = {
				planet = {
					has_modifier = standby_mode
				}
			}
			energy = 2
		}
	}

	possible_precalc = can_fill_ruler_job

	weight = {
		weight = 0
	}

	demotion = {
		time = 360
		effect = {
			pop_amount_change_category = {
				CATEGORY = civilian
				AMOUNT = local_pop_amount
			}
		}
	}

	is_capped_by_modifier = no
	can_set_priority = no
	contributes_to_diplo_weight = no
}

		default = {
			icon = unemployment
			building_icon = building_organic_sanctuary
		}
	}
	tags = { unity }

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible = {
		has_citizenship_type = { type = citizenship_organic_trophy country = owner }
		OR = {
			is_pop_category = bio_trophy
			is_pop_category = bio_trophy_unemployment
		}
		planet = { NOT = { is_planet_class = pc_cosmogenesis_world } }
	}

	resources = {
		category = unemployment_resources
	}

	weight = {
		weight = 0
	}
	is_capped_by_modifier = no
	can_set_priority = no
	contributes_to_diplo_weight = no

	promotion = {
		time = 0
		effect = {
			pop_amount_change_category = {
				CATEGORY = bio_trophy_unemployment
				AMOUNT = local_pop_amount
			}
		}
	}
}
