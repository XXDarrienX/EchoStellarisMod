@ap_biological_perfection_POINTS = 5
@ap_biological_perfection_COST_MULT = -0.1
ap_biological_perfection = {
	on_enabled = {
		add_research_option = tech_yanz_quantum_biology
		hidden_effect = {
			country_event = { id = ascensionevent.1 }
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "synthetic_empire_biological_ascension"
			NOT = { has_trait = trait_mechanical }
		}
		custom_tooltip = {
			fail_text = "requires_technology_genetic_resequencing"
			has_technology = tech_genetic_resequencing
		}
		custom_tooltip = {
			fail_text = "requires_ap_evolutionary_mastery"
			has_ascension_perk = ap_evolutionary_mastery
		}
	}
	modifier = {
		description = ap_biological_perfection_modifier_desc
		description_parameters = {
			POINTS = @ap_biological_perfection_POINTS
			COST_MULT = @ap_biological_perfection_COST_MULT
		}

		BIOLOGICAL_species_trait_points_add = @ap_biological_perfection_POINTS
		LITHOID_species_trait_points_add = @ap_biological_perfection_POINTS

		modify_species_cost_mult = @ap_biological_perfection_COST_MULT
	}

	potential = {
		host_has_dlc = "Utopia"
		NOT = {
			has_ascension_perk = ap_biological_perfection
			has_authority = "auth_machine_intelligence"
			has_authority = "auth_hive_mind"
		}
		is_mechanical_empire = no
		is_cyborg_empire = no
	}

	ai_weight = {
		factor = 100
	}
}
