@imperial_federation_cooldown = 360

# 整合度
centralization_imperium_low = {
	display_level = 1
	unlocks_centralization = 2
	cohesion_growth = 1
	potential = { has_federation_type = imperium_federation }
	allow = {
		custom_tooltip = {
			OR = {
				has_federation_law = centralization_imperium_low
				has_federation_law = centralization_imperium_mid
			}
		}
		custom_tooltip = {
			fail_text = "unified_imperium_established"
			NOT = { has_federation_flag = flag_centralization_imperium_unified_imperium }
		}
		custom_tooltip = {
			fail_text = "imperium_federation_reform_cd"
			OR = {
				has_federation_law = centralization_imperium_low
				NOR = {
					has_modifier = federation_law_category_centralization_cooldown
					has_modifier = federation_law_category_fleet_contribution_cooldown
					has_modifier = federation_law_category_succession_type_cooldown
				}
			}
		}
	}
	on_enact = {
		hidden_effect = {
			set_variable = { which = imperial_institutions_upgrades_support value = 6 }
			if = {
				limit = { has_federation_perk = imperium_elite_bureaucrats }
				change_variable = { which = imperial_institutions_upgrades_support value = 3 }
			}
			remove_modifier = federation_law_category_centralization_cooldown
			add_modifier = { modifier = federation_law_category_centralization_cooldown days = @imperial_federation_cooldown }
		}
	}
	modifier = { description = "centralization_imperium_low_effect" }
	ai_weight = {
		base = 0
		################################## 从高到低
		modifier = { # 基础支持度					+1
			desc = federation_acceptance_generally_positive
			add = 25
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { #团结度同意度修正	-100~+100	+1~-1
			desc = federation_acceptance_neg_coh_modifier
			add = value:fedcoh_neg_modifier
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 联邦团结度>0，反对降低		-1
			desc = federation_acceptance_dont_lower_centralization
			add = -25
			federation = { federation_cohesion > 0 }
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 联邦团结度<-50，支持降低		+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion < -50 }
			federation = { federation_cohesion_growth < 1.5 }
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 联邦团结度=-100，支持降低	+1
			desc = federation_acceptance_reduce_centralization
			add = 25
			federation = { federation_cohesion <= -99 }
			federation = { federation_cohesion_growth < 0 }
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 皇帝的意见					-4
			desc = federation_acceptance_emperor
			add = -100
			is_federation_leader = yes
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 皇帝的直属附庸（与议会不叠加）-1.5
			desc = federation_acceptance_emperor_vassal
			add = -37
			is_subject = yes
			overlord = { is_galactic_emperor = yes }
			is_imperial_province = no
			is_part_of_galactic_council = no
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 议会的意见					-1.5
			desc = federation_acceptance_council
			add = -37
			is_federation_leader = no
			is_part_of_galactic_council = yes
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 成员对皇帝好感度，正			-1
			desc = federation_acceptance_loyal_to_emperor
			add = -25
			is_federation_leader = no
			opinion_level = { who = galactic_emperor level >= good }
			federation = { has_federation_law = centralization_imperium_mid }
		}
		modifier = { # 成员对皇帝好感度，负			+1
			desc = federation_acceptance_disloyal_to_emperor
			add = 25
			is_federation_leader = no
			opinion_level = { who = galactic_emperor level <= poor }
			federation = { has_federation_law = centralization_imperium_mid }
		}
	}
}

				limit = { has_federation_perk = imperium_elite_bureaucrats }
				change_variable = { which = imperial_institutions_upgrades_support value = 3 }

				limit = { NOT = { has_federation_flag = flag_centralization_imperium_mid_first_time } }
