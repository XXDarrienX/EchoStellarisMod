### Name Formats

# Egalitarian
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 100
			has_government = "gov_planet_state_egalitarian"
		}
	}
	lookups = "<ps_egalitarian> <planet_state> [This.Capital.GetName]"
	noun = format.home_system
	format = format.egal.1
}
