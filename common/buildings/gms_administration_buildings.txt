
building_admincomplex_government_order = {
	base_buildtime = 400
	base_cap_amount = 1
	category = government

	potential = {
        has_upgraded_capital = yes
        exists = owner
        owner = {
            NOR = {
                has_country_flag = building_admincomplex_government_order_built
                has_country_flag = building_admincomplex_government_order_queued
            }
        }
	}

	allow = {
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
			influence = 50
		}
		produces = {
			unity = 5
			society_research = 4
		}
		upkeep = {
			energy = 2
		}
	}
	
	country_modifier = {
		edicts_cost_mult = -0.1
		edict_length_mult = 0.1
	}
	
	prerequisites = {
		"ig_tech_administration_complexes"
	}

	on_queued = { owner = { set_country_flag = building_admincomplex_government_order_queued } }
	on_unqueued = { owner = { remove_country_flag = building_admincomplex_government_order_queued } }
    on_built = { 
		owner = { 
			remove_country_flag = building_admincomplex_government_order_queued
			set_country_flag = building_admincomplex_government_order_built
		}
	}
	on_destroy = { owner = { remove_country_flag = building_admincomplex_government_order_built } }

}
