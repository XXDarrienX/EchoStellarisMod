building_planetary_chapel = {
    icon = building_planetary_chapel
	base_buildtime = 360
	planet_limit = 1
	category = unity

	building_sets = {
		unity
		urban
	}

	potential = {
		exists = owner
		owner = {
			has_active_tradition = tr_plentiful_faith_adopt
			is_gestalt = no
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				is_gestalt = yes
				NOT = { has_active_tradition = tr_plentiful_faith_adopt }
			}
		}
	}

	convert_to = {
		building_planetary_chapel_gestalt
	}
	
	allow = {
		pop_amount >= 2500
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 1
		}
	}

    planet_modifier = {
		job_faith_thinker_add = 200
	}

	triggered_desc = {
		text = job_faith_thinker_effect_desc
	}
	
	triggered_planet_modifier = {
		potential = {
		exists = owner
		owner = {
			has_active_tradition = tr_plentiful_faith_1
			}
		}
		modifier = {
			pop_government_ethic_attraction = 0.15
			planet_jobs_unity_produces_mult = 0.15
			planet_civilians_unity_produces_add = 0.5
		}	
	}
}
