building_mem_planetary_shield_generator_heavy = {
	icon = "building_mem_heavy_shield_generator"
	base_buildtime = 600
	can_build = no
	can_be_ruined = no

	category = army

	allow = {
		has_upgraded_capital = yes
	}
	
	planet_modifier = {
		planet_orbital_bombardment_damage = -0.75
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_tradition = tr_plentiful_defence_4
			}
		}
		planet_orbital_bombardment_damage = -0.1
	}

	triggered_planet_modifier = {
		potential = {
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 800
			rare_crystals = 50
		}	
		upkeep = {
			energy = 6
			rare_crystals = 1				
		}
	}

	prerequisites = {
		"mem_tech_planetary_shield_generator_heavy"
	}

	upgrades = {
		building_mem_planetary_shield_generator_orila
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}

	ai_weight = {
		weight = 10000
		modifier = {
			factor = 0
			owner = { 
				has_monthly_income = {
					resource = rare_crystals
					value < 3
				}
			}
		}
	}
}
