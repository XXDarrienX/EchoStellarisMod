#Strategic Resource Manufacturing Buildings

eb_refinery = {
	base_buildtime = @b2_time
	can_build = no

	exempt_from_ai_planet_specialization = no
	category = manufacturing

	building_sets = {
		industrial
		factory
		foundry
	}

	potential = {
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
	}

	planet_modifier = {
		planet_metallurgists_exotic_gases_produces_add = 2
		planet_artisans_exotic_gases_produces_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			industrial_jobs_are_catalytic_trigger = no
		}
		planet_metallurgists_minerals_upkeep_add = 3
		planet_artisans_minerals_upkeep_add = 3
	}


	triggered_planet_modifier = {
		potential = {
			industrial_jobs_are_catalytic_trigger = yes
		}
		planet_metallurgists_food_upkeep_add = 4
		planet_artisans_food_upkeep_add = 4
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 6
		}
	}

	prerequisites = {
		"eb_strategicresource_tech"
	}
	
	upgrades = { "eb_refinery2" }
}
