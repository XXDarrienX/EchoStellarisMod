# Queen's Den - Special Authority building for Hive Mind
building_queen_den = {
	base_buildtime = 180
	base_cap_amount = 1
	
	category = government

	potential = {
		exists = owner
		owner = { 
			is_hive_empire = yes
		}
		NOR = {
			is_resort_planet = yes
			has_modifier = slave_colony
		}
		is_capital = yes	
	}

	allow = {
		is_capital = yes
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_hive_empire = no
		}
		is_capital = no
	}

	convert_to = {
		building_primary_network_hub
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 100
		}
		upkeep = {
			energy = 3
		}
		produces = {
			unity = 10
		}
	}
		
    planet_modifier = { 
        planet_jobs_unity_produces_mult = 0.1
	}

	country_modifier = {
		planet_jobs_produces_mult = 0.1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_hive_devouring_swarm }
		}
		modifier = {
			planet_jobs_unity_produces_mult = 0.1
		}
	}
}

# Soul Harvester Building - Soul Eater Only
building_psionic_harvester = {
	base_buildtime = 180
	base_cap_amount = 1
	category = government

	potential = {
		exists = owner
		owner = {
			has_origin = civic_vb_psi_soul_eater
		}
	}

	destroy_trigger = {
		exists = owner
		NOT = {
			owner = {
				has_origin = civic_vb_psi_soul_eater
			}
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 250
		}
	}
		
    planet_modifier = { 
		job_soul_harvester_add = 2
		pop_purge_speed = -0.2
	}
}
