# Archaeotech
# For plexian clone vats: If machine empire, add scientist jobs and/or archeoresearcher jobs, unless you have the catalytic processing civic, in which case it will provide mechanical pop assembly.
building_volgan_clone_vats = {
	base_buildtime = @b2_time
	planet_limit = 1

	category = pop_assembly

	building_sets = {
		government
		urban
	}

	potential = {
		exists = owner
		owner = { has_technology = tech_secrets_volgan }
		NOT = { has_modifier = resort_colony }
		OR = {
			owner = { is_ai = no }
			NAND = {
				free_district_slots = 0
				free_building_slots <= 1
			}
		}
	}

	allow = {
		has_major_upgraded_capital = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			AND = {
				owner = { is_ai = yes }
				free_district_slots = 0
				free_building_slots = 0
			}
			has_modifier = resort_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			minor_artifacts = 75
		}
		# Base upkeep
		upkeep = {
			energy = 5
		}
		# Org/Lith upkeep
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = no }
				OR = {
					owner = { is_machine_empire = no }
					owner = { has_civic = civic_machine_servitor }
				}
			}
			food = 10
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = yes }
			}
			minerals = 10
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			NOT = { owner = { has_civic = civic_machine_servitor } }
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			OR = {
				owner = { is_machine_empire = no }
				owner = { has_civic = civic_machine_servitor }
			}
		}
		modifier = {
			logistic_growth_mult = 0.3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			OR = {
				owner = { is_machine_empire = no }
				owner = { has_civic = civic_machine_servitor }
			}
			owner = { has_ascension_perk = ap_archaeoengineers }
		}
		modifier = {
			logistic_growth_mult = 0.2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			NOT = { owner = { has_civic = civic_machine_servitor } }
		}
		modifier = {
			planet_researchers_society_research_produces_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
			NOT = { owner = { has_civic = civic_machine_servitor } }
			owner = { has_ascension_perk = ap_archaeoengineers }
		}
		modifier = {
			planet_researchers_unity_produces_add = 0.5
		}
	}

	prerequisites = {
		"tech_secrets_volgan"
	}
}
