# Archaeotech
# For plexian clone vats: If machine empire, add scientist jobs and/or archeoresearcher jobs, unless you have the catalytic processing civic, in which case it will provide mechanical pop assembly.
building_volgan_clone_vats = {
	base_buildtime = @b2_time
	planet_limit = 1

	category = amenity
	
	building_sets = {
		government
		urban
	}
	
	potential = {
		exists = owner
		owner = { has_technology = tech_secrets_volgan }
		owner = { is_wilderness_empire = no }
		owner = {
			OR = {
				is_regular_empire = yes
				is_hive_empire = yes
			}
		}
		NOT = {
			has_modifier = slave_colony
			has_modifier = penal_colony
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_machine_empire = yes
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			minor_artifacts = 75
		}
		# Base upkeep
		upkeep = {
			energy = 20
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			pop_healthcare_bonus_workforce_mult = 0.20
		}
		mult = owner.value:UCmod_archaeotech_value
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			pop_healthcare_bonus_workforce_mult = 0.05
		}
		mult = owner.value:UCmod_archaeotech_value
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		text = job_healthcare_effect_desc
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			job_healthcare_add = 100
		}
		mult = owner.value:UCmod_archaeotech_value
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_hive_empire = yes
			}
		}
		modifier = {
			pop_spawning_drone_bonus_workforce_mult = 0.05
		}
		mult = owner.value:UCmod_archaeotech_value
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_hive_empire = yes
			}
		}
		modifier = {
			job_spawning_drone_add = 100
		}
	}
	
	prerequisites = {
		"tech_secrets_volgan"
	}
	
	show_tech_unlock_if = {
		is_wilderness_empire = no
		is_machine_empire = no
	}
}

#Volgan building for machines. Decreases robot upkeep and provides power depending on the climate of the planet. Reduces habitability if placed on non-cold worlds.
