# ASTONOMICAN
building_psychic_beacon = {
	base_buildtime = @b4_time
	base_cap_amount = 1
	
	prerequisites = { "tech_light_in_darkness" }
	
	potential = {
		exists = owner
		owner = { is_spiritualist = yes }
		NOT = { has_modifier = slave_colony }
		planet = { is_capital = yes }
	}
	
	allow = {
		has_major_upgraded_capital = yes
	}
	
	destroy_trigger = {
		OR = {
			AND = {
				exists = owner
				owner = { is_regular_empire = no }
			}
			planet = { is_capital = no }
			buildings_no_zro = yes
		}
	}
	
	country_modifier = {
		country_unity_produces_mult = 0.10
	}
	
	planet_modifier = {
		job_astropath_add = 1
	}

	resources = {
		category = planet_branch_office_buildings
		cost = {
			minerals = 5000
			influence = 200
			sr_zro = @b2_rare_cost
		}	
		upkeep = {
			energy = 100
			rare_crystals = 5
		}
	}
	
	ai_weight = {
		weight = 100
	}
}

# Hyper-Cathedral
building_void_hyper_cathedral = {
	base_buildtime = @b4_time
	base_cap_amount = 1

	category = unity

	potential = {
		exists = owner
		owner = { is_spiritualist = yes }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		has_fully_upgraded_capital = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_spiritualist = no }
			NOT = { owner = { has_country_flag = dark_side } }
			has_modifier = slave_colony
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				AND = {
					NOT = { has_valid_civic = civic_death_cult_corporate }
					NOT = { has_valid_civic = civic_death_cult }
				}
			}
		}
		modifier = {
			job_priest_add = 20
			job_clerk_add = 5
			job_high_priest_add = 5
			pop_ethic_spiritualist_attraction_mult = 0.20
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				OR = {
					has_valid_civic = civic_death_cult_corporate
					has_valid_civic = civic_death_cult
				}
			}
		}
		modifier = {
			job_death_priest_add = 15
			job_high_priest_add = 5
			job_mortal_initiate_add = 10
			pop_ethic_spiritualist_attraction_mult = 0.20
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 1500
			rare_crystals = 200
		}
		upkeep = {
			energy = 50
			rare_crystals = 4
		}
	}

	prerequisites = {
		"tech_veneration_of_the_void"
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			buildings_has_crystals = yes
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value > 8
				}
			}
		}
	}
}
