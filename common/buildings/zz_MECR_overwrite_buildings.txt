building_sentinel_posts = {
	base_buildtime = @b1_time

	category = government

	building_sets = {
		government
		urban
	}

	potential = {
		exists = owner
		owner = { is_gestalt = yes }
		owner = { is_wilderness_empire = no }
		if = { #
			limit = { has_global_flag = flag_rogue_maid_mod_activated }
			owner = {
				OR = {
					is_servitor_empire = no
					servitor_empire_can_use_government_buildings = yes
				}
			}
		}
	}

	allow = {
		hidden_trigger = {
			OR = {
				owner = { is_ai = no }
				AND = {
					NOT = { has_building = building_sentinel_posts }
					NOT = { has_forbidden_jobs = patrol_drone }
					NOT = { has_available_jobs = patrol_drone }
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		if = { #
			limit = { has_global_flag = flag_rogue_maid_mod_activated }
			owner = {
				OR = {
					is_gestalt = no
					AND = {
						is_servitor_empire = yes
						servitor_empire_can_use_government_buildings = no
					}
				}
			}
		}
		else = {
			owner = { is_gestalt = no }
		}
	}

	planet_modifier = {
		job_patrol_drone_add = 200
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	prerequisites = {
		"tech_planetary_government"
	}

	triggered_desc = {
		text = job_patrol_drone_effect_desc
	}

	show_tech_unlock_if = {
		is_gestalt = yes
	}

	upgrades = { #
		building_galpol_sentinel_posts
	}
}
