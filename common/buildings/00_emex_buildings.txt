	base_buildtime = @b1_time
	planet_limit = 1

	category = amenity

	building_sets = {

		urban
	}

		NOR = {	
			has_modifier = slave_colony

		}
		exists = owner
		owner = {
			is_regular_empire = yes
			OR = {
				has_valid_civic = civic_world_council
				has_valid_civic = civic_hive_council
			}
		}
	}

	allow = {
		#buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		owner = {
			NOR = {
				has_valid_civic = civic_world_council
				has_valid_civic = civic_hive_council
			}
		}
	}

	planet_modifier = {
		planet_jobs_unity_produces_mult = 0.05
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_world_council }
		}
		modifier = {
			job_world_councillor_add = 100
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_hive_council }
		}
		modifier = {
			job_hive_councillor_add = 100
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_valid_civic = civic_world_council }
		}
		text = job_world_councillor_effect_desc
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_valid_civic = civic_hive_council }
		}
		text = job_hive_councillor_effect_desc
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}

	upgrades = {
		"building_world_council_2"
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 100
			owner = { has_valid_civic = civic_world_council }
		}
		modifier = {
			factor = 75
			owner = { has_valid_civic = civic_hive_council }
		}
