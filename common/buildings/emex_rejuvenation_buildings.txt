# Born of Knowledge - Capital Buildings
#Stage 1 - Core Processor Complex
building_bok_capital_1 = {
	base_buildtime = @b2_time
	capital = yes
	can_build = no
	can_demolish = no
	can_be_ruined = no
	can_be_disabled = no
	position_priority = 0

	category = government

	potential = {
		exists = owner
		owner = {
			AND = {
				has_origin = origin_born_of_knowledge
				is_machine_empire = yes
			}
		}
		is_capital = yes
	}

	destroy_trigger = {
		exists = owner
		owner = {
			NOT = { has_origin = origin_born_of_knowledge }
		}
	}

	convert_to = {
		building_hive_capital
		building_colony_shelter
		building_ancient_control_center
		building_ancient_palace
	}

	allow = {
		num_pops >= 1
	}

	prerequisites = {
		"tech_planetary_government"
	}

	#show_tech_unlock_if = {
	#	has_origin = origin_born_of_knowledge
	#}

	planet_modifier = {
		planet_housing_add = 3
		planet_amenities_add = 5
		job_replicator_add = 1
		job_patrol_drone_add = 1
		job_maintenance_drone_add = 2
		planet_max_buildings_add = 1
		#Addition bonuses:
		job_core_custodian_add = 2 #add
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_machine_assimilator }
		}
		modifier = {
			job_replicator_add = -1
			job_maintenance_drone_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_capital_productivity_1 }
		}
		modifier = {
			planet_jobs_productive_upkeep_mult = 0.1
			planet_jobs_productive_produces_mult = 0.1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
		}
		upkeep = {
			energy = @b2_upkeep
		}
	}

	upgrades = {
		building_bok_capital_2
	}

	triggered_desc = {
		text = job_replicator_effect_desc
	}

	triggered_desc = {
		text = job_patrol_drone_effect_desc
	}

	triggered_desc = {
		text = job_maintenance_drone_effect_desc
	}
}

# Biomatter Research Plant
building_biomatter_research = {
	base_buildtime = @b1_time
	category = resource
	base_cap_amount = 5

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			is_planet_class = pc_emex_processing_core
		}
	}

	potential = {
		NOT = { has_modifier = resort_colony }
		is_planet_class = pc_emex_bio_mechanical
	}

	planet_modifier = {
		planet_researchers_upkeep_mult = -0.02
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			food = 10
			energy = 3
		}
		produces = {
			physics_research = 1
			society_research = 1
			engineering_research = 1
		}
	}

	ai_weight = {
		weight = 2
	}
}

# Overlord Gene Labs
holding_overlord_gene_labs = {
	base_buildtime = @b1_time
	owner_type = subject_holding
	icon = building_disease_control
	base_cap_amount = 1

	category = research

	potential = {
		owner.overlord = {
			has_valid_civic = civic_hive_manipulators_of_life
		}
	}

	resources = {
		category = planet_holdings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
		produces = {
			society_research = 5
		}
	}

	planet_modifier = {
		pop_growth_speed = -0.05
		monthly_loyalty = -0.10
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet.owner
			planet.owner = { is_gestalt = no }
		}
		job_researcher_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet.owner
			planet.owner = {
				has_authority = auth_hive_mind
			}
		}
		job_brain_drone_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet.owner
			planet.owner = {
				has_authority = auth_machine_intelligence
			}
		}
		job_calculator_add = 1
	}
}
