# Eye of Terror Focuses

resolution_galactic_focus_crisis_eye_of_terror_1 = {
	icon = "GFX_resolution_emergency_measure"

	resources = {
		category = resolutions
		cost = {
			influence = 100
		}
	}
	target = no

	level = 1

	effect = {
		custom_tooltip = resolution_galactic_focus_crisis_eye_of_terror_1_tooltip

		hidden_effect = {
			set_global_flag = galactic_focus_crisis_eot_1

			set_global_flag = galactic_community_resolution_passed

			set_timed_country_flag = { flag = galactic_community_resolution_passed_general days = 360  }
			add_modifier = {
				modifier = resolution_passed_diplomatic_weight
				days = 2200
			}
		}
	}

	modifier = {
		resolution_galactic_focus_crisis_eye_of_terror_1 = 1
	
	}
	fail_effects = {
		hidden_effect = {
			set_timed_country_flag = { flag = galactic_community_resolution_failed_general days = 360  }
			add_modifier = {
				modifier = resolution_failed_diplomatic_weight
				days = 2200
			}
		}
	}

	potential = {
		has_global_flag = eye_of_terror_open		
		NOT = { has_global_flag = galactic_focus_crisis_eot_1 }
	}
		
	allow = {
		custom_tooltip = {
			fail_text = "resolution_already_active"
			NOR = {
				is_active_resolution = "resolution_galactic_focus_crisis_eye_of_terror_1"
				is_active_resolution = "resolution_galactic_focus_crisis_eye_of_terror_2"
				is_active_resolution = "resolution_galactic_focus_crisis_eye_of_terror_3"
				
			}
		}
	}	

	active = { # Resolution only in effect while this trigger is met
		
	}

	ai_weight = {
		base = 6

		modifier = {
			factor = 10

			OR = { 
				has_country_flag = had_world_devoured
				has_country_flag = had_world_invaded_by_cult
				has_country_flag = eye_of_terror_spawned_in_empire
			}
		}
		
		modifier = {
			factor = 10

			has_ascension_perk = ap_defender_of_the_galaxy
		}

		modifier = {
			factor = 0.8
			is_militarist = yes
		}

		modifier = {
			factor = 3
			is_xenophile = yes
		}

		modifier = {
			factor = 3
			is_pacifist = yes
		}

		modifier = {
			factor = 0.5
			is_xenophobe = yes
		}
	}
}
