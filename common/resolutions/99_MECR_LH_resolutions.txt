resolution_form_galactic_lower_house = {
	icon = "GFX_resolution_form_galactic_lower_house"
	resources = {
		category = resolutions
		cost = {
			influence = @resolution_cost_t4
		}
	}
	target = no

	modifier = {
		diplo_weight_mult = 0.2
		country_base_influence_produces_add = -1
	}

	effect = {
		custom_tooltip = "resolution_form_galactic_lower_house_effect"
		hidden_effect = {
			if = {
				limit = { NOT = { has_global_flag = form_galactic_lower_house_by_event } }
				set_global_flag = form_galactic_lower_house_by_resolution
				event_target:global_event_country = {
					# 启用时间表局势
					if = {
						limit = {
							NOT = { any_situation = { is_situation_type = situation_galactic_lower_house_schedule } }
						}
						start_situation = {
							type = situation_galactic_lower_house_schedule
							target = this
						}
					}
					every_situation = {
						limit = { is_situation_type = situation_galactic_lower_house_schedule }
						export_trigger_value_to_variable = {
							trigger = situation_progress
							variable = situation_progress_var
						}
						multiply_variable = { which = situation_progress_var value = -1 }
						add_situation_progress = situation_progress_var
						add_situation_progress = 65.625
					}
					every_galcom_member = {
						country_event = {
							id = MECR_LH.1010
						}
					}
					country_event = {
						id = MECR_LH.10
						days = 30
					}
					set_global_flag = lower_house_founded
					set_timed_global_flag = {
						flag = lower_house_form_succeed_timer
						years = 20
					}
					set_saved_date = {
						key = lower_house_form_succeed_date
						days_from_present = 7200
						expires = 7200
					}
				}
			}
		}
	}
	fail_effects = {
		custom_tooltip = "resolution_form_galactic_lower_house_fail_effect"
		hidden_effect = {
			event_target:global_event_country = {
				set_timed_global_flag = {
					flag = lower_house_form_failed_timer
					years = 20
				}
				set_saved_date = {
					key = lower_house_form_failed_date
					days_from_present = 7200
					expires = 7200
				}
				every_galcom_member = {
					country_event = {
						id = MECR_LH.1011
					}
				}
			}
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = "must_wait_for_20_years_to_try_form_lower_house"
			NOT = { has_global_flag = lower_house_repeal_succeed_timer }
		}
		custom_tooltip = {
			fail_text = "must_wait_for_20_years_to_try_form_lower_house_again"
			NOT = { has_global_flag = lower_house_form_failed_timer }
		}
		custom_tooltip = {
			fail_text = "the_lower_house_has_been_founded"
			NOT = { has_global_flag = lower_house_founded }
		}
	}

	potential = {
		OR = {
			has_global_flag = lower_house_form_failed
			has_global_flag = lower_house_founded
		}
	}

	ai_weight = {
		weight = @resolution_weight_popular

		modifier = {
			factor = 1.5
			is_xenophile = yes
			desc = ethic_xenophile
		}
		modifier = {
			factor = 1.5
			is_egalitarian = yes
			desc = ethic_egalitarian
		}
		modifier = {
			factor = 0.7
			is_xenophobe = yes
			desc = ethic_xenophobe
		}
		modifier = {
			factor = 0.7
			is_authoritarian = yes
			desc = ethic_authoritarian
		}
		modifier = {
			factor = value:lower_house_form_effect_relative_power_resolution
			desc = gal_com_relative_power_within_countries
			is_galactic_custodian = no
			is_galactic_emperor = no
		}
	}
}
