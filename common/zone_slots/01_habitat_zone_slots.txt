slot_habitat_02 = {
	included_zone_sets = {
		habitat_urban
	}
	unlock = { # planet scope
		custom_tooltip = {
			fail_text = zone_unavailable_tech_habitat_2
			OR = {
				AND = {
					is_capital = yes
					exists = owner
					owner = {
						is_void_dweller_empire = yes
					}
				}
				AND = {
					exists = owner
					owner = {
						has_tech_or_functional_civic = {
							TECH = tech_habitat_2
						}
					}
				}
			}
		}
	}
}
