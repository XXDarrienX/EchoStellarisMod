# Growing Pains Stage V Mutations, granted by biocrisis.230

# System Drain
scripted_action = {
	scope = self

	button_visible = {
		owner = {
			has_country_flag = behemoth_drain
		}
	}

	button_clickable = {
		custom_tooltip = {
			owner = {
				has_country_flag = behemoth_drain
			}
		}
	}

	name = on_behemoth_drain_activated
	tooltip = behemoth_drain_ship_button
	icon = GFX_fleet_action_button_behemoth_damage
	icon_selected = GFX_fleet_action_button_behemoth_damage
	activity_key = behemoth_drain_activity_key
	context_menu_name = behemoth_drain_menu_title
	cooldown = @behemoth_drain_cooldown

	wait = no

	cost = {
		food = @behemoth_drain_cost
	}
	shortcut = "SHIFT+C"
	slot = 3
}
