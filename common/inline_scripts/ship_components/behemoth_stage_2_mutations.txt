# Turbo Mode
scripted_action = {
	scope = self

	button_visible = {
		owner = {
			has_country_flag = behemoth_turbo
		}
	}

	button_clickable = {
		custom_tooltip = {
			owner = {
				has_country_flag = behemoth_turbo
			}
		}
	}

	name = on_behemoth_turbo_activated
	tooltip = behemoth_turbo_ship_button
	icon = GFX_fleet_action_button_behemoth_speed
	icon_selected = GFX_fleet_action_button_behemoth_speed
	activity_key = behemoth_turbo_activity_key
	context_menu_name = behemoth_turbo_menu_title
	cooldown = @behemoth_turbo_cooldown

	cost = {
		food = @behemoth_turbo_cost
	}
	shortcut = "SHIFT+X"
	slot = 2
}
