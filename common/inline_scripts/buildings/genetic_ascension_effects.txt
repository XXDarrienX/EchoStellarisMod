# Cloning
# Extra Clone Vats

triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			has_tradition = tr_cloning_finish
		}
	}
	building_clone_vats_max = $TIER$
    # Since we get +1 from the Cloning Tech, we reduce this by 1
    # so that the limit on Cloning Vats on a given planet is equal to the capital tier
    building_clone_vats_max = -1
}
