		$TRIGGER$_$CATEGORY$ = yes

					NOT = { has_country_flag = gave_relic_$RELIC_1$ }
				}
			}
			AND = {
				has_country_flag = selected_relic_$RELIC_2$
				event_target:curator_enclave_country = {

				event_target:curator_enclave_country = {

				}
			}
		}
	}
	name = $EVENT_ID$.3.no_relic.resources

	}
