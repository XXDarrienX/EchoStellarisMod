on_queued = {
	if = {
		limit = { is_variable_set = local_spent_biomass }
		from = {
			global_biomass_spender = yes
		}
	}
}

on_unqueued = {
	if = {
		limit = { is_variable_set = local_spent_biomass }
		from = {
			random_owned_planet = {
				limit = {
					is_habitable_for_wilderness = yes
				}
				while = {
					count = local_spent_biomass
					create_pop = {
						species = root.from
						size = 1
					}
				}
			}
		}
	}
}
