on_queued = {
	if = {
		limit = { is_variable_set = local_spent_biomass }
		from = {
			global_biomass_spender = yes
		}
	}
}

on_unqueued = {
	if = {
		limit = { is_variable_set = local_spent_biomass }
		from = {
			random_owned_planet = {
				limit = {
					is_habitable_for_wilderness = yes
				}
				create_pop_group = {
					species = root.from
					size = local_spent_biomass
				}
			}
		}
	}
}
