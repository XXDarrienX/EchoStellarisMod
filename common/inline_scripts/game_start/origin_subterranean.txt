# Subterranean Inital Setup
# Flags Set:
#	* Uncapped Mining Districts
# # Zones Added:
#	* Subterranean Urbanization


if = {
	limit = {
		exists = owner
		owner = {
			is_subterranean_empire = yes
		}
	}

	set_planet_flag = uncapped_mining

	add_district_and_planet_size_if_needed_effect_amount = {
		amount = 1
		district = district_mining_uncapped
	}
	add_zone = {
		district = district_mining_uncapped
		zone = zone_subterranean_urban
	}

	# We`ve created a mining district, so reduce the count by 1
	change_variable = {
		which = mining_districts
		value = -1
	}
}
