# Lithoid Species Initial Setup
# Rural Districts are set to:
#	* 2× Generator Districts
#	* 4× Mining Districts
#	* 0× Agriculture Districts

if = {
	limit = {
		exists = owner
		owner = {
			is_lithoid = yes
		}
	}

	set_variable = {
		which = energy_districts
		value = 2
	}
	set_variable = {
		which = mining_districts
		value = 4
	}
	set_variable = {
		which = farming_districts
		value = 0
	}
}
