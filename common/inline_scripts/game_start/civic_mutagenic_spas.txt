# Mutagenic Spas Inital Setup
# Buildings Added:
#	* Mutagenic Spa for regular empires
#	* Mutagenic Permutation Pool for hive empires
#	* Hyper Lubrication Basin for machine empires

if = {
	limit = {
		exists = owner
		owner = {
			is_regular_empire = yes
			has_toxic_baths = yes
		}
	}
	add_building = {
		zone = zone_industrial
		building = building_toxic_bath
	}
}

if = {
	limit = {
		exists = owner
		owner = {
			is_hive_empire = yes
			has_toxic_baths = yes
		}
	}
	add_building = {
		zone = zone_foundry
		building = building_toxic_bath_hive
	}
}

if = {
	limit = {
		exists = owner
		owner = {
			OR = {
				is_machine_empire = yes
				is_individual_machine = yes
			}
			has_toxic_baths = yes
		}
	}
	if = {
		limit = {
			has_zone = {
				zone = zone_industrial
			}
		}
		add_building = {
			zone = zone_industrial
			building = building_toxic_bath_machine
		}
	}
	else = {
		add_building = {
			zone = zone_foundry
			building = building_toxic_bath_machine
		}
	}
}
