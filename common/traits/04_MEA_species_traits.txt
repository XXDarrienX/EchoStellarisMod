trait_deep_dwellers = {
	cost = -4
	sorting_priority = 19
	host_has_dlc = "Aquatics Species Pack"
	potential_crossbreeding_chance = 1.0
	opposites = { "trait_nonadaptive" "trait_extremely_adaptive" "trait_robust" "trait_adaptive_lithoid" }
	allowed_archetypes = { BIOLOGICAL LITHOID }
	allowed_planet_classes = {
		pc_ocean 
	}
	initial = no
	species_potential_add = {
		always = no
	}
	species_possible_remove = {
		always = no
	}
	species_possible_merge_add = {
		always = no
	}
	# trait can't be added or removed after game start

	modifier = { 
		pc_ocean_habitability = 0.25
	}
	triggered_planet_growth_habitability_modifier = {
		potential = {
			exists = planet
			planet = {
				NOT = {
					is_planet_class = pc_ocean
				}
				OR = { 
					is_planet_class = pc_panthalassic
					is_planet_class = pc_atoll
					is_planet_class = pc_storm
					is_planet_class = pc_crag
				}
			}
		} 
		pop_environment_tolerance = 0.25
	}
	triggered_planet_growth_habitability_modifier = {
		potential = {
			exists = planet
			planet = {
				NOR = { 
					is_planet_class = pc_ocean
					is_planet_class = pc_panthalassic
					is_planet_class = pc_atoll
					is_planet_class = pc_storm
					is_planet_class = pc_crag
				}
			}
		} 
		pop_environment_tolerance = -2
	}
	triggered_pop_modifier = {
		potential = {
			exists = planet
			planet = {
				OR = {
					is_planet_class = pc_ocean
					is_planet_class = pc_panthalassic
					is_planet_class = pc_atoll
					is_planet_class = pc_storm
					is_planet_class = pc_crag
				}
			}
		}
		pop_housing_usage_mult = -0.05
		planet_jobs_food_produces_mult = 0.05
		planet_jobs_minerals_produces_mult = 0.05
		planet_jobs_energy_produces_mult = 0.05
	}
	custom_tooltip_with_modifiers = trait_deep_dwellers_tooltip
}
