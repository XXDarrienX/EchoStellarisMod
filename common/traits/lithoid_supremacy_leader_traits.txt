#Zro Overloaded
leader_trait_zro_overload = {
	inline_script = {
		script = trait/icon
		CLASS = leader
		ICON = GFX_leader_trait_zro_overload
		RARITY = common
		COUNCIL = yes
		TIER = none
	}
	inline_script = {
		script = paragon/leader_resource_upkeep
		RESOURCE = sr_zro
		AMOUNT = 1
	}
	#Basic trait
	self_modifier = {
		species_leader_exp_gain = 0.5
		leader_lifespan_add = -50
	}
	#Commander Traits
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = commander
		}
		ship_shield_mult = 0.1
		weapon_type_energy_weapon_fire_rate_mult = 0.1
		army_damage_mult = 0.2
	}
	#Scientist Traits
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = scientist
		}
		ship_archaeological_site_excavation_speed_mult = 0.3
	}
	#Govenor Traits
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = official
		}
		planet_jobs_unity_produces_mult = 0.15
		planet_jobs_physics_research_produces_mult = 0.05
		planet_jobs_society_research_produces_mult = 0.05
		planet_jobs_engineering_research_produces_mult = 0.05
	}
	#Ruler Traits
	triggered_planet_modifier = {
		potential = {
			is_ruler = no
		}
		planet_jobs_productive_upkeep_mult = -0.15
		planet_jobs_productive_produces_mult = 0.15
		planet_stability_add = -10
	}
	leader_potential_add = {
		is_ruler = no
		NOT = { from = { has_ethic = "ethic_gestalt_consciousness" } }
		has_chosen_covenant_trait_non_ruler = no	#DO NOT ADD TO IMMORTALS
	}
	leader_class = { commander scientist official }
	initial = no
	randomized = no
	icon = "gfx/interface/icons/traits/leader_trait_icons/leader_trait_zro_overload.dds"
}
