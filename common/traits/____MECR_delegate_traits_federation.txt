leader_trait_federal_negotiator = {
	leader_trait_type = veteran
	inline_script = {
		script = trait/icon
		CLASS = official
		ICON = "GFX_leader_trait_federal_negotiator"
		RARITY = free_or_veteran
		COUNCIL = no
		TIER = 1
	}
	leader_potential_add = {
		OR = {
			has_paragon_dlc = no
			has_trait = subclass_official_delegate
		}
		is_assigned_to_galcom_any_job = no
	}
	triggered_federation_modifier = {
		potential = {
			OR = {
				NOT = { exists = owner.federation }
				owner.federation = {
					is_galactic_confederation = no
					is_imperium_federation = no
				}
			}
		}
		federation_experience_add = 0.33
	}
	triggered_federation_modifier = {
		potential = {
			owner.federation = {
				OR = {
					is_galactic_confederation = yes
					is_imperium_federation = yes
				}
			}
		}
		federation_experience_add = 0.2
	}
	leader_class = { official }
	selectable_weight = {
		inline_script = paragon/subclass_free_trait_weight
		inline_script = paragon/federation_weight_mult_MECR
	}
}
