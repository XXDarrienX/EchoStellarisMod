trait_UCmod_glacial = {
	cost = 0
	potential_crossbreeding_chance = 0.33
	initial = no
	randomized = no
	species_potential_add = {
		exists = from
		from = {
			AND = {
				has_origin = origin_endangered
				has_tradition = tr_genetics_resequencing
			}
		}
	}
	species_possible_add = {
		has_trait = trait_pc_arctic_preference
		from = {
			AND = {
				has_origin = origin_endangered
				has_tradition = tr_genetics_resequencing
			}
		}
	}
	species_possible_remove = {
		can_remove_beneficial_genetic_traits = yes
	}
	allowed_archetypes = { BIOLOGICAL PRESAPIENT LITHOID }
	modifier = {
		pc_arctic_habitability = 1
	}
	custom_tooltip_with_modifiers = endangered_trait_tooltip #unlike custom_tooltip, this shows in addition to the modifiers
	triggered_pop_modifier = { #won't show in trait tooltips, use custom_tooltip/custom_tooltip_with_modifiers
		potential = {
			exists = planet
			planet = {
				is_planet_class = pc_arctic
			}
		}
		pop_happiness = 0.1
		planet_jobs_produces_mult = 0.2
	}
	triggered_pop_modifier = { #won't show in trait tooltips, use custom_tooltip/custom_tooltip_with_modifiers
		potential = {
			exists = planet
			planet = {
				NOT = { has_climate = cold }
			}
		}
		pop_happiness = -0.5
		pop_amenities_usage_add = 1
	}
	# No habitability on non-cold planets
	triggered_planet_growth_habitability_modifier = {
		potential = {
			NOR = {
				has_climate = cold
			}
			is_ideal = no
		}
		pop_environment_tolerance = -1
	}
	
	slave_cost = {
		energy = 50
	}
	
	assembly_score = {
		modifier = {
			add = 1
			from = {
				OR = {
					is_planet_class = pc_arctic
				}
			}
		}

		modifier = {
			subtract = 1
			from = {
				NOR = {
					has_climate = cold
				}
				is_ideal = no
			}
		}
	}
}
