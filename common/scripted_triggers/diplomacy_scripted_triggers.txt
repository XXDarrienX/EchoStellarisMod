	custom_tooltip = has_antagonistic_ethics
	hidden_trigger = {
		OR = {
			AND = {
				OR = {
					is_materialist = yes
					is_machine_empire = yes
				}
				FROM = { is_spiritualist = yes }
			}
			AND = {
				FROM = {
					OR = {
						is_materialist = yes
						is_machine_empire = yes
					}
				}
				is_spiritualist = yes
			}
			AND = {
				FROM = { is_militarist = yes }
				is_pacifist = yes
			}
			AND = {
				is_militarist = yes
				FROM = {
					is_pacifist = yes
				}
			}
			AND = {
				OR = {
					is_xenophobe = yes
					has_valid_civic = civic_closed_collective
					any_owned_pop = { is_being_purged = yes }
				}
				FROM = {
					OR = {
						is_xenophile = yes
						is_civic_machine_servitor = yes
					}
				}
			}
			AND = {
				OR = {
					is_xenophile = yes
					is_civic_machine_servitor = yes
				}
				FROM = {
					OR = {
						is_xenophobe = yes
						has_valid_civic = civic_closed_collective

					}
				}
			}
			AND = {
				OR = {
					is_authoritarian = yes
					is_hive_empire = yes
					any_owned_pop = { is_enslaved = yes }
				}
				FROM = {
					OR = {

					}

	}
