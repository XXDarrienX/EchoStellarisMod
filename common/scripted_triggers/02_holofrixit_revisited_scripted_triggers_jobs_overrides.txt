is_organic_species = {
	species = {
		OR = {
			is_archetype = BIOLOGICAL
			is_archetype = LITHOID
			is_archetype = FIERY
			AND = {
				is_archetype = HOLOARCH
				NOR = {
					has_trait = trait_holofrixigram
					has_trait = trait_holofrixigram_2
					has_trait = trait_holofrixigram_3
				}
			}
		}
	}
}

is_robotic_species = {
	species = {
		OR = {
			is_robotic = yes
			AND = {
				is_archetype = HOLOARCH
				OR = {
					has_trait = trait_holofrixigram
					has_trait = trait_holofrixigram_2
					has_trait = trait_holofrixigram_3
				}
			}
		}
	}
}
