is_dictatorial = {
	has_authority = auth_dictatorial
}

is_autocracy = { #Power of one
	OR = {
		has_ethic = ethic_gestalt_consciousness
		has_authority = auth_imperial
		has_authority = auth_dictatorial
	}
}

is_democracy = { #Power of many
	OR = {
		has_authority = auth_democratic
		AND = {
			is_megacorp = yes
			is_egalitarian = yes
		}
	}
}
