gpm_is_planet_fallen_empire = {
	optimize_memory
	exists = owner
	owner = {
		OR = {
			is_country_type = fallen_empire
			is_country_type = awakened_fallen_empire
		}
	}
}
gpm_is_fallen_empire_space_owner = {
	optimize_memory
	exists = space_owner
	space_owner = {
		OR = {
			is_country_type = fallen_empire
			is_country_type = awakened_fallen_empire
		}
	}
}
gpm_is_planet_primitive = {
	optimize_memory
	exists = owner
	owner = {
		is_primitive = yes
	}
}
