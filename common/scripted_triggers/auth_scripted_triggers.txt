
is_megacorp = {
	optimize_memory
	OR = {
		has_authority = auth_corporate
		has_government = gov_imperial_domain_megacorp
	}
}

is_dictatorial = {
	optimize_memory
	OR = {
		has_authority = auth_dictatorial
		# Other Mods
		is_egalitarian = no
	}
}

is_imperial = { #Has Heir
	optimize_memory
	OR = {
		has_authority = auth_imperial
	}
}

	OR = {
		is_gestalt = yes
		is_imperial = yes
		is_dictatorial = yes
	}
}

is_imp_or_dict = { #Power of one
	OR = {

	}
}
