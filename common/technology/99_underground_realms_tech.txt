	area = society
	cost = @tier5cost3
	tier = 5
	is_rare = yes
	is_dangerous = yes
	is_reverse_engineerable = no
	category = { psionics }
	prerequisites = { "tech_colossus" }
	ai_update_type = military
	weight = @tier5weight3
	
	potential = {
		has_country_flag = colossus_project
		OR = {
			has_country_flag = dark_side
			has_modifier = covenant_eater_of_worlds
			has_modifier = covenant_composer_of_strands
			has_modifier = covenant_instrument_of_desire
			has_modifier = covenant_whisperers_in_the_void
			has_modifier = covenant_end_of_the_cycle
		}
	}
	
	weight_modifier = {
		factor = 0.50
		modifier = {
			factor = @ap_technological_ascendancy_rare_tech
			has_ascension_perk = ap_technological_ascendancy
		}
		modifier = {
			factor = @federation_perk_factor
			has_federation = yes
			federation = {
				has_federation_perk = rare_tech_boost
				OR = {
					any_member = { has_technology = tech_pk_shrouder }
					any_member = { has_technology = tech_pk_godray }
				}
			}
		}
		modifier = {
			factor = 0.20
			NOR = {
				research_leader = {
					area = society
					has_trait = "leader_trait_maniacal"
				}
				research_leader = {
					area = society
					has_trait = "leader_trait_curator"
				}
				research_leader = {
					area = society
					has_trait = "leader_trait_expertise_psionics"
				}
			}			
		}
		modifier = {
			factor = 1.50
			has_country_flag = dark_side
		}
	}
	
	ai_weight = {
		factor = 0.25
		modifier = {
			factor = 1.25
			has_ethic = ethic_spiritualist
		}
		modifier = {
			factor = 1.5
			has_ethic = ethic_fanatic_spiritualist
		}
	}
}
