# Resort Colony
decision_sanctuary_colony = {
	owned_planets_only = yes
	icon = decision_resort_colony
	
	prerequisites = {
		"tech_sanctuary_colonies"
	}

	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}
	
	potential = {
		is_capital = no
		NOT = { has_modifier = penal_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { is_planet_class = pc_city }
		NOT = { has_modifier = sanctuary_colony }
		owner = { 
			is_machine_empire = yes 
			has_valid_civic = civic_machine_servitor
		}
		is_artificial = no
		owner = {
			is_ai = no
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = decision_resort_colony_no_districts
			num_districts = { type = any value < 1 }
			num_buildings = { type = any value < 2 }
		}
	}
	
	effect = {
		add_modifier = {
			modifier = sanctuary_colony
			days = -1
		}	
		hidden_effect = { validate_planet_buildings_and_districts = yes }
	}
	
	ai_weight = {
		weight = 0
	}
}

decision_abolish_sanctuary_colony = {
	owned_planets_only = yes

		has_modifier = sanctuary_colony
	}

	effect = {
		remove_modifier = sanctuary_colony
		hidden_effect = { validate_planet_buildings_and_districts = yes }
	}

	}
}
