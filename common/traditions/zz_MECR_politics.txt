tr_politics_adopt = {
	possible = {
		custom_tooltip = {
			fail_text = tr_politics_adopt_not_enough_relations
			OR = {
				has_global_flag = galactic_community_founded
				has_global_flag = galcom_founding_begun
				count_relation = { # minimum 3 countries
					limit = {
						is_country_type = default
						has_communications = prev
						is_homicidal = no
					}
					count >= 2
				}
			}
		}
	}
	unlocks_agenda = agenda_pioneer_politics
	custom_tooltip = tr_politics_adopt_desc
	modifier = {
		envoys_add = 2
		#Envoys generate influence, handled in the leaders script
	}

	on_enabled = {
		hidden_effect = {
			country_event = {
				id = MECR.150
			}
		}
	}
	inline_script = MECR/traditions/MECR_swap_tr_politics_adopt	#MECR
}
