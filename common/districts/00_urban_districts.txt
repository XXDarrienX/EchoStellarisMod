@base_cost = 300
@city_cost = 500

district_city = {
	base_buildtime = 480
	is_capped_by_modifier = no

	show_on_uncolonized = {
		exists = from
		from = { is_regular_empire = yes }
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
		}
	}

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		is_resort_planet = no
		OR = {
			uses_district_set = standard
			uses_district_set = hive_world
			uses_district_set = machine_world
		}
	}

	allow = {
		is_resort_planet = no
		NOT = { has_modifier = slave_colony }
		str_city_district_allow = yes
	}

	conversion_ratio = 0.5
	convert_to = {
		district_arcology_housing
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts_cities
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 5
		planet_max_buildings_add = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			str_enable_clerks = yes 
		}
		modifier = {
			job_clerk_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_tradition = tr_plentiful_affinity_finish }
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_interstellar_franchising }
			str_enable_clerks = yes 
		}
		modifier = {
			job_clerk_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_clerk_effect_desc
	}
	ai_resource_production = {
		alloys = 10
			trigger = { 
				free_housing < 25
				num_pops > 15
				free_building_slots = 0
				nor = {
				has_designation = col_factory
				has_designation = col_foundry
				}
				or = {
				owner = {
				used_naval_capacity_percent > 0.9
				resource_stockpile_compare = {
					resource = alloys
					value > 1000
				}
				resource_income_compare = {
					resource = alloys
					value > 30
				} }
				and = { is_capital = yes free_housing < 15 }
			} 
		}
	}
	ai_resource_production = {
		alloys = 30
			trigger = { 
				str_force_city_district = yes 
			} 
	}

	ai_resource_production = {
		society_research = 50
			trigger = { 
				str_force_city_district = yes 
			} 
	}


	ai_resource_production = {
		energy =  30
			trigger = { 
				str_force_city_district = yes 
			} 
	}

	ai_resource_production = {
		alloys = 10
		energy = 10
			trigger = {
				num_pops > 15
				years_passed > 40 
				free_housing < 20
				free_building_slots = 0
				or = {
				has_designation = col_factory
				has_designation = col_foundry
				}
				free_jobs > 0
				owner = {
				used_naval_capacity_percent > 0.9
				resource_stockpile_compare = {
					resource = alloys
					value > 1000
				} 
				resource_income_compare = {
					resource = alloys
					value > 30
				}
			} 
		}
	}
}
