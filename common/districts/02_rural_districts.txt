		}

	}

		tech_power_plant_1
	}
	
	ai_weight = {
		factor = 0
		modifier = {
			weight = 2
			factor = 1
			gai_build_district = yes
		}
		modifier = {
			weight = 100
			factor = 1
			gai_build_district = yes

				owner = {

			weight = 9000
			factor = 1
			gai_good_energy = yes
			planet = { count_pops = { limit = { is_sapient = yes } count > 3 } }

			always = yes
