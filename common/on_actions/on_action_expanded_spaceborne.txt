	}
}

on_decade_pulse = {

	}
}

# Triggers event when a system is entered by ships from a country (not pirates or monsters) for the first time
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system_first_time = {
	events = {
		caraspace.1
		spacecreatures.2
	}
}

# This = owner of fleet 1 (destroyed)
