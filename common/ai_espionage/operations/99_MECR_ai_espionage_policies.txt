weaken_the_imperium_policy = {	#削弱帝国
	weight = {
		modifier = {
			MECR_is_difficult_mode_for_emperor = no
			has_galactic_emperor = yes
			target = { is_galactic_emperor = yes }
			owner = {
				OR = {
					any_owned_leader = { is_undermining_imperial_authority = yes }
					has_modifier = undermining_efforts_stalled
					opinion_level = { who = galactic_emperor level <= poor }
				}
			}
			is_spynetwork_max_level > 75
			has_spynetwork_value > 40
			weight = 70
		}
		modifier = {	# 高难：降低需求
			MECR_is_difficult_mode_for_emperor = yes
			has_galactic_emperor = yes
			target = { is_galactic_emperor = yes }
			owner = {
				OR = {
					any_owned_leader = { is_undermining_imperial_authority = yes }
					has_modifier = undermining_efforts_stalled
					opinion = { who = galactic_emperor value <= -50 }
				}
			}
			is_spynetwork_max_level > 50
			has_spynetwork_value > 35
			weight = 70
		}
	}
}
destroy_the_imperium_policy = {	#推翻帝国
	weight = {
		modifier = {
			has_galactic_emperor = yes
			target = { is_galactic_emperor = yes }
			owner = {
				OR = {
					any_owned_leader = { is_undermining_imperial_authority = yes }
					has_modifier = undermining_efforts_stalled
					opinion_level = { who = galactic_emperor level <= poor }
					AND = {
						MECR_is_difficult_mode_for_emperor = yes
						opinion = { who = galactic_emperor value <= -50 }
					}
				}
			}
			is_spynetwork_max_level > 75
			has_spynetwork_value > 40
			imperial_authority <= 50
			weight = 1000
		}
	}
}
