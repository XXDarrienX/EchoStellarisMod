defined_text = {
	name = GetScientistResearchSubroutine
	text = {
		trigger = {
			NOT = {
				has_ethic = ethic_gestalt_consciousness
			}
		}
		localization_key = MEMloc_scientist
	}
	text = {
		trigger = {
			is_hive_empire = yes
		}
		localization_key = MEMloc_res_drone
	}
	text = {
		trigger = {
			is_hive_empire = yes
		}
		localization_key = MEMloc_res_subroutine
	}
}
