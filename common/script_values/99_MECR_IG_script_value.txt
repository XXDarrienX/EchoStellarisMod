situation_governance_not_universal_size_need = {	#视为充分的治理需要：帝国拥有的星系数量*3+500
	base = 500
	complex_trigger_modifier = {
		trigger = count_system
		parameters = {
			limit = {
				exists = owner
				owner = { is_galactic_community_member = yes }
			}
		}
		mode = add
		mult = 3
	}
}

# 由建筑提供行政能力的旧代码：弃用
building_imperial_government_sv = {
	base = 0
	complex_trigger_modifier = {
		potential = { NOT = { has_global_flag = PR_installed } }
		trigger = count_owned_pop_group
		parameters = {
			limit = {
				OR = {
					has_job = priest
					has_job = death_priest
					has_job = bureaucrat
					has_job = unifier
					has_job = manager
					has_job = steward
					has_job = coordinator
					has_job = evaluator
					has_job = synapse_drone
					has_job = chronicle_drone
					has_job = head_maid #rogue maid
				}
			}
		}
		mode = add
	}
	modifier = {
		has_global_flag = PR_installed
		add = planet.value:PR_prmt_sv_plnt_JOB_efficiency|CHAIN|culture|JIC|planet_administrators|
	}
	mult = value:imperial_governance_produces_mult_sv
}
