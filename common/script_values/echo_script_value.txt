merc_defense_contracts = {
	## Number of Defensive Pacts
	complex_trigger_modifier = {
		trigger = num_defensive_pacts
		mode = add
	}
	## Number of Guarantees
	complex_trigger_modifier = {
		trigger = num_guarantees
		mode = add
	}
}

merc_company_bonus = {
	## War Counts
	complex_trigger_modifier = {
		trigger = count_war
		parameters = {
			limit = {
				any_war_participant = {
					is_country = root
					is_war_leader = no
				}
			}
		}
		mode = add
		mult = 1
	}
}

brothel_bonus = {

		mode = add
		mult = 1
	}
}

soul_diet = {
	base = 0
	complex_trigger_modifier = {
		trigger = count_owned_species
		trigger_scope = owner
		parameters = {
			limit = {
				is_robotic_species = no
				is_soul_eater = no
				has_slavery_type = { country = from type = slavery_livestock }
			}
		}
		mode = add
		mult = 1
	}
}
