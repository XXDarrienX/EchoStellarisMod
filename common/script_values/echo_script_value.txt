merc_company_bonus = {
	## War Counts
	complex_trigger_modifier = {
		trigger = count_war
		parameters = {
			limit = {
				any_war_participant = {
					is_country = root
					is_war_leader = no
				}
			}
		}
		mode = add
		mult = 1
	}
}

	base = 0
	complex_trigger_modifier = {

				is_soul_eater_empire = yes
				is_robotic_species = no

			}

		mode = add
