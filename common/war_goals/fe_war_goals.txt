wg_ae_domination = {
	casus_belli = cb_subjugation
	set_defender_wargoal = wg_end_threat_ae
	hide_if_no_cb = yes
	total_war = yes

	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}

	allowed_peace_offers = {
		status_quo
		surrender
	}
	
	cede_claims = occupied_only

	potential = {
		is_country_type = awakened_fallen_empire
		NOT = { has_ethic = ethic_gestalt_consciousness }
	}
	possible = {
		always = yes
	}
	on_accept = {
		if = {
			limit = { from = { has_been_declared_crisis = yes } }
			# do nothing;
			# FEs may go to war against BtC empires, just not vassalize them
		}
		else_if = {
			limit = {
				is_xenophile = yes
			}
			from = {
				set_subject_of = {
					who = root
					subject_type = signatory
				}				
			}	
		}
		else_if = {
			limit = {
				is_xenophobe = yes
			}
			from = {
				set_subject_of = {
					who = root
					subject_type = thrall
				}	
			}
		}
		else_if = {
			limit = {
				is_spiritualist = yes
			}
			from = {
				set_subject_of = {
					who = root
					subject_type = dominion
				}		
			}
		}
		else_if = {
			limit = { 
				is_materialist = yes
			}
			from = {
				set_subject_of = {
					who = root
					subject_type = satellite
				}
			}
		}
	}
	
	ai_weight = {
		weight = 100
	}
}

wg_end_threat_ae = {
	casus_belli = cb_containment
	
	hide_if_no_cb = yes
	total_war = yes
	set_defender_wargoal = wg_ae_domination

	allowed_peace_offers = {
		status_quo
		surrender
	}
	
	surrender_acceptance = -1000

	potential = {
		is_country_type = default
		from = { is_country_type = awakened_fallen_empire }
	}
	possible = {
		always = yes
	}
	
	ai_weight = {
		weight = 100
	}
}
