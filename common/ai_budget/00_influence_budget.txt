influence_expenditure_decisions = {
	resource = influence
	type = expenditure
	category = decisions

	potential = {
		always = yes
	}

	fraction = {
		weight = 0.20
		modifier = {
			factor = 2
			OR = {
				AND = {
					has_technology = tech_housing_2
					any_owned_planet = {
						is_planet_class = pc_relic
						num_uncleared_blockers < 1
					}
				}
				AND = {
					has_ascension_perk = ap_arcology_project
					any_owned_planet = {
						is_artificial = no
						NOR = {
							is_planet_class = pc_city
							is_planet_class = pc_relic
							has_modifier = resort_colony
							has_modifier = penal_colony
							has_modifier = slave_colony
							has_modifier = pm_wenkwort_custodian
						}
						num_uncleared_blockers < 1
						free_district_slots < 1
						OR = {
							num_pops >= 70
							is_capital = yes
						}
					}
				}
			}
		}
	}

	# this is the cost of the arcology project decision
	static_max = {
		base = 200
	}
}

influence_expenditure_megastructures = {
	resource = influence
	type = expenditure
	category = megastructures

	potential = {
		is_at_war = no
		OR = {
			has_technology = tech_gateway_construction
			has_technology = tech_ring_world
		}

		has_resource = {
			type = influence
			amount > 500
		}
	}

	fraction = {
		weight = 0.1
	}

	# only used by construction sites, base cost is 300
	static_min = {
		base = 75
		modifier = {
			add = 225
			has_technology = tech_ring_world
			NOT = {
				any_owned_planet = { is_planet_class = pc_ringworld_habitable }
			}
		}
	}
	static_max = {
		base = 75
		modifier = {
			add = 225
			has_technology = tech_ring_world
		}
	}
}
