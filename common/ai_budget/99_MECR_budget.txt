## Categories used in this file must have use_for_ai_budget = yes defined in economic_categories ##
## It is possible to have multiple entries with the same category and resource ##
alloys_expenditure_ships_MECR_imperial_armada_conscription_reduction = {
	resource = alloys
	type = expenditure
	category = ships

	potential = {
		is_galactic_community_member = yes
		has_galactic_emperor = yes
		exists = federation
		federation = {
			OR = {
				has_federation_law = imperial_armada_conscription_low
				has_federation_law = imperial_armada_conscription_mid
				has_federation_law = imperial_armada_conscription_high
			}
		}
	}

	fraction = {
		weight = 0
		modifier = {
			add = -0.3
			used_naval_capacity_percent > value:ia_keep_percent
		}
		modifier = {
			add = -0.15
			used_naval_capacity_percent > value:ia_keep_percent_1_5
		}
		modifier = {
			add = -0.15
			used_naval_capacity_percent > value:ia_keep_percent_2
		}
		modifier = {
			add = -0.15
			used_naval_capacity_percent > value:ia_keep_percent_2_5
		}
		modifier = {
			add = -0.1
			used_naval_capacity_percent > value:ia_keep_percent_3
		}
		modifier = {
			add = -0.1
			used_naval_capacity_percent > value:ia_keep_percent_3_5
		}
		modifier = {
			add = -0.1
			used_naval_capacity_percent > value:ia_keep_percent_4
		}
		
		# spend more alloys building ships while at war
		modifier = {
			factor = 2
			OR = {
				recently_lost_war = yes
				is_at_war = yes
			}
		}
		# If we want to colonize and are at peace then prio that
		modifier = {
			factor = 0.25
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
			}
		}
		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}
		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.0
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
	}
}
