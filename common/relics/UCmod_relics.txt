@activation_cost = 3000
@triumph_duration = 3600
@triumph_duration_short = 1800
@triumph_duration_veryshort = 900

r_the_frozen_engine = {
	activation_duration = @triumph_duration
	portrait = "GFX_relic_frozen_engine"
	sound = "relic_activation_the_last_baol"

	resources = {
		category = relics
		# Activation cost
		cost = {
			unity = @activation_cost
		}
	}

	ai_weight = {
		weight = 100
	}

	triggered_country_modifier = {
		potential = {
			always = yes
		}
		modifier = {
			country_energy_produces_mult = 0.2
		}
	}

	triggered_country_modifier = {
		potential = {
			has_origin = origin_endangered
		}
		modifier = {
			pop_happiness = 0.1
		}
	}

	score = 1000

	active_effect = {
		custom_tooltip = relic_frozen_engine_active_tooltip
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown"
				days = @triumph_duration
			}
			if = {
				limit = {
					NOT = { has_country_flag = relic_frozen_engine_activated }

			}
		}
	}

	# Possible check for activation
	possible = {
		inline_script = {
			script = relics/activation_checks
			RELIC = r_the_frozen_engine
		}
	}
