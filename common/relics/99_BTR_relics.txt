@activation_cost = 3000
@triumph_duration = 3600
@triumph_duration_short = 1800
@triumph_duration_veryshort = 900

r_hitori_pendant = {
	activation_duration = @triumph_duration
	portrait = "GFX_relic_hitori_pendant"
	sound = "relic_activation_hitori_pendant"

	resources = {
		category = relics
		# Activation cost
		cost = {
			unity = 500
		}
	}

	ai_weight = {
		weight = 10
	}

	triggered_country_modifier = {
		potential = {
			always = yes
		}
		country_base_unity_produces_add = 50
	}

	score = 1000

	active_effect = {
		add_modifier = {
			modifier = "rock_band_hitori"
			days = @triumph_duration_veryshort
		}
		custom_tooltip = relic_triumph_cooldown_veryshort
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown"
				days = @triumph_duration_veryshort
			}
		}
	}

	# Possible check for activation
	possible = {
		custom_tooltip = {
			fail_text = "requires_relic_no_cooldown"
			NOT = { has_modifier = relic_activation_cooldown }
		}
	}
}
