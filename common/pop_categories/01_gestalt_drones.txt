complex_drone = {
	rank = 0
	clothes_texture_index = 2

	change_job_threshold = 1.25

	weight = {
		weight = 2
	}

	pop_modifier = {
		pop_housing_usage_base = 1
		pop_amenities_usage_no_happiness_base = 1
	}

	# scope: pop to be resettled
	allow_resettlement = {
		always = yes
	}

	resettlement_cost = {
		energy = 100
	}

	resettlement_cost = {
		trigger = {
			owner = {
				NOR = {
					has_valid_civic = civic_machine_ota_updates
					has_valid_civic = civic_hive_subsumed_will
				}
			}
		}
		unity = 10
	}

	triggered_planet_modifier = {
		potential = {
			can_assemble_budding_pop = yes
		}
		planet_pop_assembly_organic_add = @plantoid_budding_rate
	}

	unemployment_resources = {
		category = unemployment_resources
	}

	resources = {
		category = pop_category_drones

		upkeep = {
			trigger = {
				has_phototrophic_energy_upkeep = yes
			}
			energy = @living_standard_phototrophic_normal
		}

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_phototrophic = no
				is_lithoid = no
				is_soul_eater = no
				is_energy_being = no
			}
			food = @living_standard_food_normal
		}

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_phototrophic = yes
				is_lithoid = no
			}
			food = @living_standard_phototrophic_normal
		}

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_phototrophic = no
				is_lithoid = yes
			}
			minerals = @living_standard_food_normal
		}

		upkeep = {
			trigger = {
				is_organic_species = yes
				is_phototrophic = yes
				is_lithoid = yes
			}
			minerals = @living_standard_phototrophic_normal
		}

		upkeep = {
			trigger = {
				is_robotic_species = yes
			}
			energy = @living_standard_energy_normal
		}

		# Exotic Metabolism Upkeep
		upkeep = {
			trigger = {
				is_organic_species = yes
				has_trait = trait_exotic_metabolism
			}
			exotic_gases = @exotic_metabolism
		}

		produces = {
			trigger = {
				is_lithoid = yes
				OR = {
					has_trait = trait_lithoid_scintillating
					has_trait = trait_monocrystal_geodes
				}
			}
			rare_crystals = @lithoid_trait_resource_production
		}

		produces = {
			trigger = {
				is_lithoid = yes
				OR = {
					has_trait = trait_lithoid_gaseous_byproducts
					has_trait = trait_monocrystal_geodes
				}
			}
			exotic_gases = @lithoid_trait_resource_production
		}

		produces = {
			trigger = {
				is_lithoid = yes
				OR = {
					has_trait = trait_lithoid_volatile_excretions
					has_trait = trait_monocrystal_geodes
				}
			}
			volatile_motes = @lithoid_trait_resource_production
		}
		#Energy Being Upkeep
		upkeep = {
			trigger = { 
				is_energy_being = yes
			}
			energy = @living_standard_food_normal
		}
		#Soul Eater Upkeep
		upkeep = {
			trigger = { 
				is_soul_eater = yes
			}
			soul_crystals = @living_standard_food_normal
		}
		### Lithoid Supremacy content
		#Dune Born
		produces = {
			trigger = { 
				is_lithoid = yes
				has_trait = trait_dune_born
				planet = {
					is_planet_class = pc_desert
			}
		}
			volatile_motes = 0.02
		}

		#Mossy Symbiosis
		upkeep = {
			trigger = { 
				is_organic_species = yes
				is_lithoid = yes
				has_trait = trait_mossy_symbiosis
			}
			minerals = -0.25
			food = 0.25
		}
		produces = {
			trigger = { 
				is_organic_species = yes
				has_trait = trait_inorganic_breath
			}
			exotic_gases = @inorganic_breath
		}
	}
}
