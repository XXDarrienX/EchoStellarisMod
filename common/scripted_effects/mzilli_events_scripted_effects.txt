apply_frail_fusion_effect = {
	if = {
		limit = {
			OR = {
				is_colonizable = yes
				exists = owner
			}
			NOR = {
				is_artificial = yes
				is_planet_class = pc_city
				is_planet_class = pc_machine
			}
		}
		set_planet_flag = frail_fusion
		if = {
			limit = { NOT = { has_modifier = bleak_planet } }
			add_modifier = { modifier = bleak_planet days = -1 }
		}
	}
}
