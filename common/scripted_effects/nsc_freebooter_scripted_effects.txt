origin_nsc_freebooter_start_setup = {
	set_planet_entity = {
		entity = "habitat_phase_03_entity"
		graphical_culture = space_owner
	}
	solar_system = {
		every_system_planet = {
			limit = { has_planet_flag = mining_planet }
			create_mining_station = { owner = space_owner }
		}
	}
	#add_deposit = d_nsc_freebooter_home #disable this code for now
	add_deposit = d_nsc_freebooter_habitat
	while = {
		count = 28 #changed to match that of the vanilla
		create_pop = { species = owner.species }
	}
	add_building = building_capital
	add_building = building_bureaucratic_1
	add_building = building_research_lab_1
	add_building = building_foundry_1
	while = {
		count = 2
		add_district = district_nsc_freebooter_hab_housing
	}
	while = {
		count = 1
		add_district = district_nsc_freebooter_hab_industrial
	}
	if = {
		limit = { owner = { is_lithoid_empire = no } }
		while = { add_district = district_nsc_freebooter_hab_farming count = 2 }
		else = { while = { add_district = district_nsc_freebooter_hab_mining count = 2 } }
	}
	add_district = district_nsc_freebooter_hab_mining
}
