create_hibernators_pops = {
	custom_tooltip = create_hibernators_pops_ct
	hidden_effect = {
		create_species = {
			name = random
			class = random_pre_ftl
			homeworld = this
			portrait = random
			traits = {
				trait = trait_seasonal_dormancy
			}
			traits = random
			sapient = no
			effect = {
				save_event_target_as = hibernators_species
			}
		}
		while = {
			count = @hibernators_pops_created
			create_pop_group = {
				species = event_target:hibernators_species
			}
		}
	}
}
