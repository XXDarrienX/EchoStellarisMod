calculate_defense_contractor_bonus = {
	remove_modifier = uncivil_defence_contractor_mod
	
	export_trigger_value_to_variable = {
		trigger = num_defensive_pacts
		variable = merc_defense_pacts
	}
	multiply_variable = {
		which = merc_defense_pacts
		value = merc_defense_total
	}
	add_modifier = {
		modifier = uncivil_defence_contractor_mod
		multiplier = merc_defense_total
		days = -1
	}
	clear_variable = merc_defense_pacts
	clear_variable = merc_defense_total
}

calculate_merc_war_bonus = {
	remove_modifier = uncivil_mercenary_company_mod
	export_trigger_value_to_variable = {
		trigger = count_war
		parameters = {
			any_war_participant = {
				is_country = root
				is_war_leader = no }
		}
		variable = merc_wars
	}
	multiply_variable = {
		which = merc_wars
		value = merc_war_total
	}
	add_modifier = {
		modifier = uncivil_mercenary_company_mod
		multiplier = merc_war_total
		days = -1
	}
	clear_variable = merc_war_total
	clear_variable = merc_wars
}
