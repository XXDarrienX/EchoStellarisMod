# Precursor Selection Effects
##################################

FHMP_MAP_Precursor_Selection_Effect = {
	capital_scope = {
		solar_system = {
			while = {
				count = 40
				closest_system = {
					limit = {
						NOR = {
							has_star_flag = $FLAG$
							has_star_flag = empire_home_system
						}
					}
					set_star_flag = $FLAG$
				}
			}
		}
	}
}

FHMP_increment_num_precursors_found = {
	if = {
		limit = { is_variable_set = PreSelect_num_precursors_found }
		subtract_variable = {
			which = PreSelect_num_precursors_found
			value = 1
		}
	}
}
