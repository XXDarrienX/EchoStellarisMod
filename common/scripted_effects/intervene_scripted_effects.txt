# Get Intervention Effects
get_flag_war_effect = {
	every_war = {
		every_war_participant = {
			limit = {
				is_war_leader = yes
				is_same_value = prevprev	
			}
			prev = { set_war_flag = war_friendly_nation_@prev }
		}
		prev = {
			change_variable = {
				which = friendly_nation_war_count
				value = 1
			}
		}
	}
}
