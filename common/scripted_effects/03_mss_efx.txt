ms_swap_to_maid_portrait = {
    if = {
        limit = { is_robotic = yes }
        random_list = {
            1 = { change_species_characteristics = { portrait = ms_robotic_01 } }
            1 = { change_species_characteristics = { portrait = ms_robotic_02 } }
            1 = { change_species_characteristics = { portrait = ms_human_01_machine } }
        }
    }
    else = {
        switch = {
