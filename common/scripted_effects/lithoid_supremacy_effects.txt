#Lithoid Supremacy Effects

add_leader_traits_after_modification_lithoid_supremacy = {
	#Add Sandstone Hewn (Official)
	if = {
		limit = {
			species = {
				has_trait = trait_sandstone_hewn
			}
		}
		if = {
			limit = {
				leader_class = official
				NOT = { has_trait = leader_trait_official_sandstone }
			}
			add_trait = leader_trait_official_sandstone
		}
	}
	#Add Chroma Stones (Commander)
	if = {
		limit = { 
			species = {
				has_trait = trait_chroma_stones
			}
		}
		if = {
			limit = {
				leader_class = commander
				NOT = { has_trait = leader_trait_commander_chroma_stones }
			}
			add_trait = leader_trait_commander_chroma_stones
		}
	}
	#Add Internal Ballistics (Commander)
	if = {
		limit = { 
			species = {
				has_trait = trait_internal_ballistics
			}
		}
		if = {
			limit = {
				leader_class = commander
				NOT = { has_trait = leader_trait_commander_internal_ballistics }
			}
			add_trait = leader_trait_commander_internal_ballistics
		}
	}
	#Add Zro Infused
	if = {
		limit = {
			species = {
				has_trait = trait_zro_infused
			}
		}
		if = {
			limit = {
				NOR = {
					has_trait = leader_trait_zro_infused
					has_leader_flag = renowned_leader
					has_leader_flag = legendary_leader
				}
			}
			add_trait_no_notify = leader_trait_zro_infused
		}
	}
}

remove_leader_traits_after_modification_lithoid_supremacy = {
	#Remove Sandstone Hewn (Official)
	if = {
		limit = {
			species = {
				NOT = { has_trait = trait_sandstone_hewn }
			}
		}
		if = {
			limit = {
				has_trait = leader_trait_official_sandstone
			}
			remove_trait = leader_trait_official_sandstone
		}
	}
	#Remove Chroma Stones (Commander)
	if = {
		limit = { 
			species = {
				NOT = { has_trait = trait_chroma_stones }
			}
		}
		if = {
			limit = {
				leader_class = commander
				has_trait = leader_trait_commander_chroma_stones
			}
			remove_trait = leader_trait_commander_chroma_stones
		}
	}
	#Remove Internal Ballistics (Commander)
	if = {
		limit = { 
			species = {
				NOT = { has_trait = trait_internal_ballistics }
			}
		}
		if = {
			limit = {
				leader_class = commander
				has_trait = leader_trait_commander_internal_ballistics
			}
			remove_trait = leader_trait_commander_internal_ballistics
		}
	}
	#Remove Zro Infused
	if = {
		limit = {
			species = {
				OR = {
					NOT = { has_trait = trait_zro_infused }
					has_trait = trait_cybernetic
				}
			}
		}
		if = {
			limit = { has_trait = leader_trait_zro_infused }
			remove_trait = leader_trait_zro_infused
		}
	}
}
