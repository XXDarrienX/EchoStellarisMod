	category = deposit_cat_rare

	potential = { always = no }

	#The old cost
    #resources = {
	#	category = planet_deposits

    #        sr_dark_matter = -5
	#	}
	#}

	#Scaling Costs
	resources = {
		category = planet_deposits
		produces = {
			trigger = { planet = { planet_size > 20 } }
			energy = -50
            sr_dark_matter = -5
		}
		produces = {
			trigger = {
				planet = { planet_size > 15 }
				planet = { planet_size < 21 }
			}
			energy = -40
            sr_dark_matter = -4
		}
		produces = {
			trigger = {
				planet = { planet_size > 10 }
				planet = { planet_size < 16 }
			}
			energy = -30
            sr_dark_matter = -3
		}
		produces = {
			trigger = {
				planet = { planet_size > 5 }
				planet = { planet_size < 11 }
			}
			energy = -20
            sr_dark_matter = -2
		}
		produces = {
			trigger = {
				planet = { planet_size > 0 }
				planet = { planet_size < 5 }
			}
			energy = -10
            sr_dark_matter = -1
		}
	}

		weight = 0
	}
}
